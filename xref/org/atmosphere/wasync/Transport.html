18  /**
19   * Define a supported {@link Transport}. Default are websocket, streaming, server-side events and long-polling.
20   * Use this interface only if you want to define another transport. New transport can be registered using the {@link Options}
21   * class.
22   *
23   * @author Jeanfrancois Arcand
24   */

28       * The transport name

35       * @param function {@link FunctionResolver}
36       * @return this;
37       */
38      Transport registerF(FunctionWrapper function);
39  
40      /**
41       * Called when an unexpected exception ocurred.
42       * @param t a {@link Throwable}
43       */
44      void onThrowable(Throwable t);
45  
46      /**
47       * Close the underlying transport}
48       */
49      void close();
50  
51      /**
52       * Return the current {@link org.atmosphere.wasync.Socket.STATUS}
53       */
54      Socket.STATUS status();
55  
56      /**
57       * Error handled by a {@link Function}
58       */
59      boolean errorHandled();
60  
61      void error(Throwable e);
62  }
