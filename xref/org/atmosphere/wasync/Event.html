16  package org.atmosphere.wasync;
17  
18  /**
19   *  This class define the type of event a {@link Function} or {@link Decoder} will be called with.
20   *
21   *  @author Jeanfrancois Arcand
22   */
23  public enum Event {
24  
25      /**
26       * This event is fired when the connection is open. This event is only fired once.
27       */
28      OPEN,
29  
30      /**
31       * This event is fired when the connection gets closed. This event is only fired once.
32       */
33      CLOSE,
34  
35      /**
36       * This event is fired every time a new message is received.
37       */
38      MESSAGE,
39  
40      /**
41       * This event is fired every time the connection re-opened
42       */
43      REOPENED,
44  
45      /**
46       * This event is fired when unexpected error happens. This event is only fired once.
47       */
48      ERROR,
49  
50      /**
51       * This event is fire when the response's status is received. This event is only fired once.
52       */
53      STATUS,
54  
55      /**
56       * This event is fire when the response's headers is received. This event is only fired once.
57       */
58      HEADERS,
59  
60      /**
61       * This event is fire when the response's bytes are received.
62       */
63      MESSAGE_BYTES,
64  
65      /**
66       * This event is fire when the connection has been established. The Transport propagated is the one that worked.
67       * This event is only fired once.
68       */
69      TRANSPORT
70  
71  }
