16  package org.atmosphere.wasync;
17  
18  /**
19   * An asynchronous client's implementation used to create {@link Socket} and {@link Request}. As simple as as
20   * <blockquote><pre>
21       Client client = ClientFactory.getDefault().newClient();
22  
23       RequestBuilder request = client.newRequestBuilder()
24               .method(Request.METHOD.GET)
25               .uri(targetUrl + "/suspend")
26               .decoder(new Decoder&lt;String, POJO&gt;() {
27                   &#64;Override
28                   public POJO decode(String s) {
29                       return new POJO(s);
30                   }
31               })
32               .transport(Request.TRANSPORT.WEBSOCKET);
33   * </pre></blockquote>
34   * @author Jeanfrancois Arcand
35   */
36  public interface Client<O extends Options, U extends OptionsBuilder, T extends RequestBuilder> {
37  
38      /**
39       * Create a {@link Socket}
40       *
41       * @return {@link Socket}
42       */
43      Socket create();
44  
45      /**
46       * Create a {@link Socket} configured using the {@link Options}
47       *
48       * @return {@link Socket}
49       */
50      Socket create(O options);
51  
52      /**
53       * Return a {@link RequestBuilder}
54       *
55       * @return a {@link RequestBuilder}
56       */
57      T newRequestBuilder();
58  
59      /**
60       * Create a new {@link RequestBuilder} based on the class' implementation.
61       *
62       * @param clazz an implementation of {@link RequestBuilder}
63       * @return a {@link RequestBuilder}
64       */
65      T newRequestBuilder(Class<T> clazz);
66  
67  
68      /**
69       * Return an {@link OptionsBuilder}
70       *
71       * @return {@link OptionsBuilder}
72       */
73      U newOptionsBuilder();
74  }
