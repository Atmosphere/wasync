45  /**
46   * {@link InputStream} implementation that reads a character stream from a {@link Reader}
47   * and transforms it to a byte stream using a specified charset encoding. The stream
48   * is transformed using a {@link CharsetEncoder} object, guaranteeing that all charset
49   * encodings supported by the JRE are handled correctly. In particular for charsets such as
50   * UTF-16, the implementation ensures that one and only one byte order marker
51   * is produced.
52   * <p>
53   * Since in general it is not possible to predict the number of characters to be read from the
54   * {@link Reader} to satisfy a read request on the {@link ReaderInputStream}, all reads from
55   * the {@link Reader} are buffered. There is therefore no well defined correlation
56   * between the current position of the {@link Reader} and that of the {@link ReaderInputStream}.
57   * This also implies that in general there is no need to wrap the underlying {@link Reader}
58   * in a {@link java.io.BufferedReader}.
59   * <p>
60   * {@link ReaderInputStream} implements the inverse transformation of {@link java.io.InputStreamReader};
61   * in the following example, reading from <tt>in2</tt> would return the same byte
62   * sequence as reading from <tt>in</tt> (provided that the initial byte sequence is legal
63   * with respect to the charset encoding):
64   * <pre>
65   * InputStream in = ...
66   * Charset cs = ...
67   * InputStreamReader reader = new InputStreamReader(in, cs);
68   * ReaderInputStream in2 = new ReaderInputStream(reader, cs);</pre>
69   * {@link ReaderInputStream} implements the same transformation as {@link java.io.OutputStreamWriter},
70   * except that the control flow is reversed: both classes transform a character stream
71   * into a byte stream, but {@link java.io.OutputStreamWriter} pushes data to the underlying stream,
72   * while {@link ReaderInputStream} pulls it from the underlying stream.
73   * <p>
74   * Note that while there are use cases where there is no alternative to using
75   * this class, very often the need to use this class is an indication of a flaw
76   * in the design of the code. This class is typically used in situations where an existing
77   * API only accepts an {@link InputStream}, but where the most natural way to produce the data
78   * is as a character stream, i.e. by providing a {@link Reader} instance. An example of a situation
79   * where this problem may appear is when implementing the {@link javax.activation.DataSource}
80   * interface from the Java Activation Framework.
81   * <p>
82   * Given the fact that the {@link Reader} class doesn't provide any way to predict whether the next
83   * read operation will block or not, it is not possible to provide a meaningful
84   * implementation of the {@link InputStream#available()} method. A call to this method
85   * will always return 0. Also, this class doesn't support {@link InputStream#mark(int)}.
86   * <p>
87   * Instances of {@link ReaderInputStream} are not thread safe.
88   *  *
89   * @since 2.0
90   */
91  public class ReaderInputStream extends InputStream {
92      private static final int DEFAULT_BUFFER_SIZE = 1024;
93  
94      private final Reader reader;
95      private final CharsetEncoder encoder;
96  
97      /**
98       * CharBuffer used as input for the decoder. It should be reasonably
99       * large as we read data from the underlying Reader into this buffer.
100      */
101     private final CharBuffer encoderIn;
102 
103     /**
104      * ByteBuffer used as output for the decoder. This buffer can be small
105      * as it is only used to transfer data from the decoder to the
106      * buffer provided by the caller.
107      */
108     private final ByteBuffer encoderOut;
109 
110     private CoderResult lastCoderResult;
111     private boolean endOfInput;
112 
113     /**

115      *
116      * @param reader the target {@link Reader}
117      * @param encoder the charset encoder
118      * @since 2.1
119      */
120     public ReaderInputStream(Reader reader, CharsetEncoder encoder) {
121         this(reader, encoder, DEFAULT_BUFFER_SIZE);

124     /**
125      * Construct a new {@link ReaderInputStream}.
126      *
127      * @param reader the target {@link Reader}
128      * @param encoder the charset encoder
129      * @param bufferSize the size of the input buffer in number of characters
130      * @since 2.1
131      */
132     public ReaderInputStream(Reader reader, CharsetEncoder encoder, int bufferSize) {
133         this.reader = reader;
134         this.encoder = encoder;
135         this.encoderIn = CharBuffer.allocate(bufferSize);
136         this.encoderIn.flip();
137         this.encoderOut = ByteBuffer.allocate(128);
138         this.encoderOut.flip();
139     }
140 
141     /**
142      * Construct a new {@link ReaderInputStream}.
143      *
144      * @param reader the target {@link Reader}
145      * @param charset the charset encoding
146      * @param bufferSize the size of the input buffer in number of characters
147      */
148     public ReaderInputStream(Reader reader, Charset charset, int bufferSize) {
149         this(reader,
150              charset.newEncoder()
151                     .onMalformedInput(CodingErrorAction.REPLACE)
152                     .onUnmappableCharacter(CodingErrorAction.REPLACE),
153              bufferSize);
154     }

199     /**
200      * Fills the internal char buffer from the reader.
201      *
202      * @throws IOException
203      *             If an I/O error occurs
204      */
205     private void fillBuffer() throws IOException {
206         if (!endOfInput && (lastCoderResult == null || lastCoderResult.isUnderflow())) {
207             encoderIn.compact();
208             int position = encoderIn.position();
209             // We don't use Reader#read(CharBuffer) here because it is more efficient
210             // to write directly to the underlying char array (the default implementation
211             // copies data to a temporary char array).
212             int c = reader.read(encoderIn.array(), position, encoderIn.remaining());
213             if (c == -1) {
214                 endOfInput = true;
215             } else {
216                 encoderIn.position(position+c);
217             }
218             encoderIn.flip();
219         }
220         encoderOut.compact();
221         lastCoderResult = encoder.encode(encoderIn, encoderOut, endOfInput);
222         encoderOut.flip();
223     }
224 
225     /**
226      * Read the specified number of bytes into an array.
227      *
228      * @param b the byte array to read into
229      * @param off the offset to start reading bytes into
230      * @param len the number of bytes to read
231      * @return the number of bytes read or <code>-1</code>
232      *         if the end of the stream has been reached
233      * @throws IOException if an I/O error occurs
234      */
235     @Override
236     public int read(byte[] b, int off, int len) throws IOException {
237         if (b == null) {
238             throw new NullPointerException("Byte array must not be null");
239         }
240         if (len < 0 || off < 0 || (off + len) > b.length) {
241             throw new IndexOutOfBoundsException("Array Size=" + b.length +
242                     ", offset=" + off + ", length=" + len);
243         }
244         int read = 0;
245         if (len == 0) {
246             return 0; // Always return 0 if len == 0
247         }
248         while (len > 0) {
249             if (encoderOut.hasRemaining()) {
250                 int c = Math.min(encoderOut.remaining(), len);
251                 encoderOut.get(b, off, c);
252                 off += c;
253                 len -= c;
254                 read += c;
255             } else {
256                 fillBuffer();
257                 if (endOfInput && !encoderOut.hasRemaining()) {
258                     break;
259                 }
260             }
261         }
262         return read == 0 && endOfInput ? -1 : read;
263     }
264 
265     /**
266      * Read the specified number of bytes into an array.
267      *
268      * @param b the byte array to read into
269      * @return the number of bytes read or <code>-1</code>
270      *         if the end of the stream has been reached
271      * @throws IOException if an I/O error occurs
272      */
273     @Override
274     public int read(byte[] b) throws IOException {
275         return read(b, 0, b.length);
276     }
277 
278     /**
279      * Read a single byte.
280      *
281      * @return either the byte read or <code>-1</code> if the end of the stream
282      *         has been reached
283      * @throws IOException if an I/O error occurs
284      */
285     @Override
286     public int read() throws IOException {
287         for (;;) {
288             if (encoderOut.hasRemaining()) {
289                 return encoderOut.get() & 0xFF;
290             } else {
291                 fillBuffer();
292                 if (endOfInput && !encoderOut.hasRemaining()) {
293                     return -1;
294                 }
295             }
296         }
297     }
298 
299     /**
300      * Close the stream. This method will cause the underlying {@link Reader}
301      * to be closed.
302      * @throws IOException if an I/O error occurs
303      */
304     @Override
305     public void close() throws IOException {
306         reader.close();
307     }
