19  import com.ning.http.client.AsyncHttpClientConfig;

28  import org.slf4j.LoggerFactory;

30  import java.io.IOException;
31  import java.util.concurrent.CountDownLatch;
32  import java.util.concurrent.TimeUnit;
33  import java.util.concurrent.atomic.AtomicLong;

35  /**
36   * A utility class that can be used to load a WebSocket enabled Server
37   *
38   * @author jeanfrancois Arcand
39   */
40  public class WebSocketLoader {
41  
42      private final static Logger logger = LoggerFactory.getLogger(WebSocketLoader.class);
43  
44      public static void main(String[] s) throws InterruptedException, IOException {
45  
46          if (s.length == 0) {
47              s = new String[]{"5", "1", "1", "http://127.0.0.1:8080/simple/test"};
48          }
49  
50          int run = Integer.valueOf(s[0]);
51          final int clientNum = Integer.valueOf(s[1]);
52          final int messageNum = Integer.valueOf(s[2]);
53          String url = s[3];
54  
55          System.out.println("Stressing: " + url);
56          System.out.println("Number of Client: " + clientNum);
57          System.out.println("Number of Message: " + messageNum);
58          System.out.println("Number of run: " + run);
59          long count = 0;
60  
61          for (int r = 0; r < run; r++) {
62  
63              AsyncHttpClientConfig.Builder b = new AsyncHttpClientConfig.Builder();
64              b.setFollowRedirect(true).setConnectTimeout(-1).setReadTimeout(-1).setUserAgent("loader/1.1");
65  
66              NettyAsyncHttpProviderConfig nettyConfig = new NettyAsyncHttpProviderConfig();
67  
68              nettyConfig.addProperty("child.tcpNoDelay", "true");
69              nettyConfig.addProperty("child.keepAlive", "true");
70  
71              final AsyncHttpClient c = new AsyncHttpClient(b.setAsyncHttpClientProviderConfig(nettyConfig).build());
72              Client client = ClientFactory.getDefault().newClient();
73              RequestBuilder request = client.newRequestBuilder();
74              request.method(Request.METHOD.GET).uri(url);
75              request.transport(Request.TRANSPORT.WEBSOCKET);
76              request.header("X-wakeUpNIO", "true");
77  
78              final CountDownLatch l = new CountDownLatch(clientNum);
79              final CountDownLatch messages = new CountDownLatch(messageNum * clientNum);
80              long clientCount = l.getCount();
81              final AtomicLong total = new AtomicLong(0);
82  
83              Socket[] sockets = new Socket[clientNum];
84              for (int i = 0; i < clientCount; i++) {
85                  final AtomicLong start = new AtomicLong(0);
86                  sockets[i] = client.create(client.newOptionsBuilder().runtime(c).reconnect(false).build())
87                          .on(new Function<Integer>() {
88                              @Override
89                              public void on(Integer statusCode) {
90                                  start.set(System.currentTimeMillis());
91                                  l.countDown();
92                              }
93                          }).on(new Function<String>() {
94  
95                              int mCount = 0;
96  
97                              @Override
98                              public void on(String s) {
99                                  if (s.startsWith("message")) {
100                                     String[] m = s.split("\n\r");
101                                     mCount += m.length;
102                                     messages.countDown();
103                                     System.out.println("Message left receive " + messages.getCount() + " message " + s);
104                                     if (mCount == messageNum) {
105                                         // System.out.println("All messages received " + mCount);
106                                         total.addAndGet(System.currentTimeMillis() - start.get());
107                                     }
108                                 }
109                             }
110                         }).on(new Function<Throwable>() {
111                             @Override
112                             public void on(Throwable t) {
113                                 t.printStackTrace();
114                             }
115                         });
116 
117             }
118 
119             for (int i = 0; i < clientCount; i++) {
120                 sockets[i].open(request.build());
121             }
122 
123             l.await(30, TimeUnit.SECONDS);
124 
125             System.out.println("OK, all Connected: " + clientNum);
126 
127             Socket socket = client.create(client.newOptionsBuilder().runtime(c).build());
128             socket.open(request.build());
129             for (int i = 0; i < messageNum; i++) {
130                 socket.fire("message" + i);
131             }
132             messages.await(1, TimeUnit.HOURS);
133             socket.close();
134             for (int i = 0; i < clientCount; i++) {
135                 sockets[i].close();
136             }
137             count += (total.get() / clientCount);
138             System.out.println("Run " + r + " => Total run : " + (total.get() / clientCount));
139             c.close();
140             System.gc();
141         }
142         System.out.println("=== Means " + (count/run) + "=====");
143     }
144
