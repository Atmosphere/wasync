16  package org.atmosphere.wasync.impl;
17  
18  import com.ning.http.client.AsyncHttpClient;
19  import com.ning.http.client.AsyncHttpClientConfig;
20  import org.atmosphere.wasync.Options;
21  import org.atmosphere.wasync.Socket;
22  
23  public class ClientUtil {
24      private static final String WASYNC_USER_AGENT = "wAsync/1.0";
25  
26      public final static AsyncHttpClient createDefaultAsyncHttpClient(Options o) {
27  		AsyncHttpClientConfig.Builder b = new AsyncHttpClientConfig.Builder();
28  		b.setFollowRedirects(true).setRequestTimeoutInMs(o.requestTimeout()).setUserAgent(WASYNC_USER_AGENT);
29          AsyncHttpClientConfig config = b.build();
30  		return new AsyncHttpClient(config);
31  	}
32  
33      public static Socket create(Options options) {
34          AsyncHttpClient asyncHttpClient = options.runtime();
35          if (asyncHttpClient == null || asyncHttpClient.isClosed()) {
36              asyncHttpClient = ClientUtil.createDefaultAsyncHttpClient(options);
37              options.runtime(asyncHttpClient);
38          }
39          return getSocket(options);
40      }
41  
42      public final static Socket getSocket(Options options) {
43          return new DefaultSocket(options);
44      }
45  
46      public  static Socket create() {
47          AsyncHttpClient asyncHttpClient = createDefaultAsyncHttpClient(new DefaultOptionsBuilder().reconnect(true).build());
48  
49          return getSocket(new DefaultOptionsBuilder().runtime(asyncHttpClient, false).build());
50      }
