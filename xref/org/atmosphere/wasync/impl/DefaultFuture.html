16  package org.atmosphere.wasync.impl;
17  
18  import org.atmosphere.wasync.Future;
19  import org.atmosphere.wasync.Socket;
20  
21  import java.io.IOException;
22  import java.util.concurrent.CountDownLatch;
23  import java.util.concurrent.ExecutionException;
24  import java.util.concurrent.TimeUnit;
25  import java.util.concurrent.TimeoutException;
26  import java.util.concurrent.atomic.AtomicBoolean;
27  
28  public class DefaultFuture implements Future {
29  
30      private final Socket socket;
31      private final CountDownLatch latch = new CountDownLatch(1);
32      private final AtomicBoolean done = new AtomicBoolean(false);
33  
34      public DefaultFuture(Socket socket) {
35          this.socket = socket;
36      }
37  
38      @Override
39      public boolean cancel(boolean mayInterruptIfRunning) {
40          latch.countDown();
41          return true;
42      }
43  
44      @Override
45      public boolean isCancelled() {
46          return latch.getCount() == 0;
47      }
48  
49      @Override
50      public boolean isDone() {
51          return done.get();
52      }
53  
54      // TODO: Not public
55      public Future done(){
56          done.set(true);
57          latch.countDown();
58          return this;
59      }
60  
61      @Override
62      public Socket get() throws InterruptedException, ExecutionException {
63          latch.await();
64          return socket;
65      }
66  
67      @Override
68      public Socket get(long timeout, TimeUnit unit) throws InterruptedException, ExecutionException, TimeoutException {
69          latch.await(timeout, unit);
70          return socket;
71      }
72  
73      protected Socket socket() {
74          return socket;
75      }
76  
77      public Future fire(Object data) throws IOException {
78          ((DefaultSocket)socket).internalSocket().write(((DefaultSocket)socket).request(), data);
79          return new DefaultFuture(socket);
80      }
81  }
