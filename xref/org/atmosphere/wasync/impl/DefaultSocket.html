77      public Socket open(Request request) throws IOException {
78          this.request = request;
79          RequestBuilder r = new RequestBuilder();
80          r.setUrl(request.uri())
81                  .setMethod(request.method().name())
82                  .setHeaders(request.headers());
83  
84          List<Transport> transports = getTransport(request);
85  
86          return connect(r, transports);
87      }
88  
89      protected Socket connect(RequestBuilder r,  List<Transport> transports) throws IOException {
90  
91          if (transports.size() > 0) {
92              transportInUse = transports.remove(0);
93          } else {
94              throw new IOException("No suitable transport supported");
95          }
96  
97          Future f = new Future(this);
98          transportInUse.future(f);
99          if (transportInUse.name().equals(Request.TRANSPORT.WEBSOCKET)) {
100             r.setUrl(request.uri().replace("http", "ws"));
101             try {
102                 java.util.concurrent.Future<WebSocket> w = asyncHttpClient.prepareRequest(r.build()).execute(
103                         (AsyncHandler<WebSocket>) transportInUse);
104 
105                 socket = new InternalSocket(w.get());
106             } catch (ExecutionException t) {
107                 Throwable e = t.getCause();
108                 if (e != null) {
109                     if (e.getMessage() != null && e.getMessage().equalsIgnoreCase("Invalid handshake response")) {
110                         logger.info("WebSocket not supported, downgrading to an HTTP based transport.");
111                         return connect(r, transports);
112                     }
113                 }
114                 transportInUse.onThrowable(t);
115                 return new VoidSocket();
116             } catch (Throwable t) {
117                 transportInUse.onThrowable(t);
118                 return new VoidSocket();
119             }
120         } else {
121             r.setUrl(request.uri().replace("ws", "http"));
122             java.util.concurrent.Future<String> s = asyncHttpClient.prepareRequest(r.build()).execute(
123                     (AsyncHandler<String>) transportInUse);
124 
125             try {
126                 // TODO: Give a chance to connect and then unlock. With Atmosphere we will received junk at the
127                 // beginning for streaming and sse, but nothing for long-polling
128                 f.get(2500, TimeUnit.MILLISECONDS);
129             } catch (Throwable t) {
130                 // Swallow  LOG ME
131             }
132 
133             socket = new InternalSocket(asyncHttpClient);
134         }
135         return this;
136     }
137 
138     @Override
139     public void close() {
140         if (socket != null) {
141             socket.close();
142             transportInUse.close();
143         }
144     }

146     protected List<Transport> getTransport(Request request) throws IOException {
147         List<Transport> transports = new ArrayList<Transport>();
148 
149         for (Request.TRANSPORT t : request.transport()) {
150             if (t.equals(Request.TRANSPORT.WEBSOCKET)) {
151                 transports.add(new WebSocketTransport(request.decoders(), functions, request.functionResolver()));
152             } else if (t.equals(Request.TRANSPORT.SSE)) {
153                 transports.add(new SSETransport(request.decoders(), functions, request.functionResolver()));
154             } else if (t.equals(Request.TRANSPORT.LONG_POLLING)) {
155                 transports.add(new LongPollingTransport(request.decoders(), functions, request, asyncHttpClient, request.functionResolver()));
156             } else if (t.equals(Request.TRANSPORT.STREAMING)) {
157                 transports.add(new StreamTransport(request.decoders(), functions, request.functionResolver()));
158             }
159         }
160         return transports;
161     }

164     private final static class InternalSocket {

166         private final WebSocket webSocket;

171             this.asyncHttpClient = null;

182             } else {
183                 asyncHttpClient.close();
184             }

187         Object invokeEncoder(List<Encoder<? extends Object, ?>> encoders, Object instanceType) {
188             for (Encoder e : encoders) {
189                 Class<?>[] typeArguments = TypeResolver.resolveArguments(e.getClass(), Encoder.class);
190 
191                 if (typeArguments.length > 0 && typeArguments[0].equals(instanceType.getClass())) {
192                     instanceType = e.encode(instanceType);
193                 }
194             }
195             return instanceType;
196         }

198         public InternalSocket write(Request request, Object data) throws IOException {
199 
200             // Execute encoder
201             Object object = invokeEncoder(request.encoders(), data);
202             if (webSocket != null) {
203                 if (InputStream.class.isAssignableFrom(object.getClass())) {
204                     InputStream is = (InputStream) object;
205                     ByteArrayOutputStream bs = new ByteArrayOutputStream();
206                     //TODO: We need to stream directly, in AHC!
207                     byte[] buffer = new byte[8192];
208                     int n = 0;
209                     while (-1 != (n = is.read(buffer))) {
210                         bs.write(buffer, 0, n);
211                     }
212                     webSocket.sendMessage(bs.toByteArray());
213                 } else if (Reader.class.isAssignableFrom(object.getClass())) {
214                     Reader is = (Reader) object;
215                     StringWriter bs = new StringWriter();
216                     //TODO: We need to stream directly, in AHC!
217                     char[] chars = new char[8192];
218                     int n = 0;
219                     while (-1 != (n = is.read(chars))) {
220                         bs.write(chars, 0, n);
221                     }
222                     webSocket.sendTextMessage(bs.getBuffer().toString());
223                 } else if (String.class.isAssignableFrom(object.getClass())) {
224                     webSocket.sendTextMessage(object.toString());
225                 } else if (byte[].class.isAssignableFrom(object.getClass())) {
226                     webSocket.sendMessage((byte[]) object);
227                 } else {
228                     throw new IllegalStateException("No Encoder for " + data);
229                 }
230             } else {
231                 if (InputStream.class.isAssignableFrom(object.getClass())) {
232                     //TODO: Allow reading the response.
233                     asyncHttpClient.preparePost(request.uri())
234                             .setMethod(Request.METHOD.POST.name())
235                             .setBody((InputStream) object).execute();
236                 } else if (Reader.class.isAssignableFrom(object.getClass())) {
237                     asyncHttpClient.preparePost(request.uri())
238                             .setMethod(Request.METHOD.POST.name())
239                             .setBody(new ReaderInputStream((Reader) object))
240                             .execute();
241                     return this;
242                 } else if (String.class.isAssignableFrom(object.getClass())) {
243                     asyncHttpClient.preparePost(request.uri())
244                             .setMethod(Request.METHOD.POST.name())
245                             .setBody((String) object)
246                             .execute();
247                 } else if (byte[].class.isAssignableFrom(object.getClass())) {
248                     asyncHttpClient.preparePost(request.uri())
249                             .setMethod(Request.METHOD.POST.name())
250                             .setBody((byte[]) object)
251                             .execute();
252                 } else {
253                     throw new IllegalStateException("No Encoder for " + data);
254                 }
255             }
256             return this;
257         }
258     }
