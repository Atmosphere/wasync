74                  transportInUse.status().equals(STATUS.ERROR)) {
75              transportInUse.error(new IOException("Invalid Socket Status " + transportInUse.status().name()));

106     @Override
107     public Socket open(Request request, long timeout, TimeUnit tu) throws IOException {
108         this.request = request;
109         RequestBuilder r = new RequestBuilder();
110         r.setUrl(request.uri())
111                 .setMethod(request.method().name())
112                 .setHeaders(request.headers())
113                 .setQueryParameters(decodeQueryString(request));
114 
115         List<Transport> transports = getTransport(r, request);
116 
117         return connect(r, transports, timeout, tu);
118     }

120     static FluentStringsMap decodeQueryString(Request request) {
121         Map<String, List<String>> c = request.queryString();
122         FluentStringsMap f = new FluentStringsMap();
123         f.putAll(c);
124         return f;
125     }

131     protected Socket connect(final RequestBuilder r, final List<Transport> transports, final long timeout, final TimeUnit tu) throws IOException {
132 
133         if (transports.size() > 0) {
134             transportInUse = transports.get(0);
135         } else {
136             throw new IOException("No suitable transport supported");
137         }
138         socketRuntime = new SocketRuntime(options, new DefaultFuture(this), functions);
139 
140         functions.add(new FunctionWrapper("", new Function<TransportNotSupported>() {
141             @Override
142             public void on(TransportNotSupported transportNotSupported) {
143                 request.transport().remove(0);
144                 if (request.transport().size() > 0) {
145                     try {
146                         if (request.queryString().get("X-Atmosphere-Transport") != null) {
147                             Request.TRANSPORT rt = request.transport().get(0);
148                             String t = rt == Request.TRANSPORT.LONG_POLLING ? "long-polling" : rt.name();
149                             request.queryString().put("X-Atmosphere-Transport", Arrays.asList(new String[]{t}));
150                         }
151                         open(request, timeout, tu);
152                     } catch (IOException e) {
153                         logger.error("", e);
154                     }
155                 } else {
156                     throw new Error("No suitable transport supported by the server");
157                 }
158             }
159         }));
160 
161         if (transportInUse.name().equals(Request.TRANSPORT.WEBSOCKET)) {
162             r.setUrl(request.uri().replace("http", "ws"));
163             try {
164                 java.util.concurrent.Future<WebSocket> fw = options.runtime().prepareRequest(r.build()).execute(
165                         (AsyncHandler<WebSocket>) transportInUse);
166 
167                 fw.get(timeout, tu);
168                 socketRuntime = new SocketRuntime(WebSocketTransport.class.cast(transportInUse), options, socketRuntime.future(), functions);
169             } catch (ExecutionException t) {
170                 Throwable e = t.getCause();
171 
172                 if (TransportNotSupported.class.isAssignableFrom(e.getClass())) {
173                     return this;
174                 }
175 
176                 transportInUse.close();
177                 closeRuntime(true);
178                 if (!transportInUse.errorHandled() && TimeoutException.class.isAssignableFrom(e.getClass())) {
179                     transportInUse.error(new IOException("Invalid state: " + e.getMessage()));
180                 }
181 
182                 return new VoidSocket();
183             } catch (Throwable t) {
184                 transportInUse.onThrowable(t);
185                 return new VoidSocket();
186             }
187         } else {
188             r.setUrl(request.uri().replace("ws", "http"));
189             options.runtime().prepareRequest(r.build()).execute((AsyncHandler<String>) transportInUse);
190 
191             try {
192                 // TODO: Give a chance to connect and then unlock. With Atmosphere we will received junk at the
193                 // beginning for streaming and sse, but nothing for long-polling
194                 socketRuntime.future().get(options.waitBeforeUnlocking(), TimeUnit.MILLISECONDS);
195             } catch (Throwable t) {
196                 // Swallow  LOG ME
197                 logger.trace("", t);
198             }
199 
200             socketRuntime = createSocket();
201         }
202         return this;
203     }

209     /**
210      * {@inheritDoc}
211      */
212     @Override
213     public void close() {
214         // Not connected, but close the underlying AHC.
215         if (transportInUse == null) {
216             closeRuntime(false);
217         } else if (socketRuntime != null && !transportInUse.status().equals(STATUS.CLOSE)) {
218             transportInUse.close();
219             closeRuntime(true);
220         }
221     }
222 
223     void closeRuntime(boolean async) {
224         if (!options.runtimeShared() && !options.runtime().isClosed()) {
225             if (async) {
226                 // AHC is broken when calling closeAsynchronously.
227                 // https://github.com/AsyncHttpClient/async-http-client/issues/290
228                 final ExecutorService e= Executors.newSingleThreadExecutor();
229                 e.submit(new Runnable() {
230                     @Override
231                     public void run() {
232                         options.runtime().close();
233                         e.shutdown();
234                     }
235                 });
236             }
237             else
238                 options.runtime().close();
239         } else if (options.runtimeShared()) {
240             logger.warn("Cannot close underlying AsyncHttpClient because it is shared. Make sure you close it manually.");
241         }
242     }

257     protected List<Transport> getTransport(RequestBuilder r, Request request) throws IOException {
258         List<Transport> transports = new ArrayList<Transport>();
259 
260         if (request.transport().size() == 0) {
261             transports.add(new WebSocketTransport(r, options, request, functions));
262             transports.add(new LongPollingTransport(r, options, request, functions));
263         }
264 
265         for (Request.TRANSPORT t : request.transport()) {
266             if (t.equals(Request.TRANSPORT.WEBSOCKET)) {
267                 transports.add(new WebSocketTransport(r, options, request, functions));
268             } else if (t.equals(Request.TRANSPORT.SSE)) {
269                 transports.add(new SSETransport(r, options, request, functions));
270             } else if (t.equals(Request.TRANSPORT.LONG_POLLING)) {
271                 transports.add(new LongPollingTransport(r, options, request, functions));
272             } else if (t.equals(Request.TRANSPORT.STREAMING)) {
273                 transports.add(new StreamTransport(r, options, request, functions));
274             }
275         }
276         return transports;
277     }
