16  package org.atmosphere.wasync.impl;
17  
18  import org.atmosphere.wasync.Client;
19  import org.atmosphere.wasync.FunctionResolver;
20  import org.atmosphere.wasync.OptionsBuilder;
21  import org.atmosphere.wasync.RequestBuilder;
22  import org.atmosphere.wasync.Socket;
23  
24  /**
25   * The default implementation of the {@link Client}
26   *
27   * @author Jeanfrancois Arcand
28   */
29  public class DefaultClient implements Client<DefaultOptions, OptionsBuilder,RequestBuilder> {
30  
31      public DefaultClient() {
32      }
33  
34      /**
35       * {@inheritDoc}
36       */
37      @Override
38      public Socket create() {
39          return ClientUtil.create();
40      }
41  
42      /**
43       * {@inheritDoc}
44       */
45      @Override
46      public Socket create(DefaultOptions options) {
47          return ClientUtil.create(options);
48      }
49  
50      /**
51       * {@inheritDoc}
52       */
53      @Override
54      public RequestBuilder newRequestBuilder() {
55          RequestBuilder b = new DefaultRequestBuilder();
56          return b.resolver(FunctionResolver.DEFAULT);
57      }
58  
59      /**
60       * {@inheritDoc}
61       */
62      @Override
63      public OptionsBuilder newOptionsBuilder() {
64          return new DefaultOptionsBuilder();
65      }
66  
67      /**
68       * {@inheritDoc}
69       */
70      @Override
71      public RequestBuilder newRequestBuilder(Class<RequestBuilder> clazz) {
72          RequestBuilder b = null;
73          try {
74              b = clazz.newInstance();
75          } catch (InstantiationException e) {
76              throw new RuntimeException(e);
77          } catch (IllegalAccessException e) {
78              throw new RuntimeException(e);
79          }
80          return b.resolver(FunctionResolver.DEFAULT);
81      }
82  }
