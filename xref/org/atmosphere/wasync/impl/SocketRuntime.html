69      protected Object invokeEncoder(List<Encoder<? extends Object, ?>> encoders, Object instanceType) {
70          for (Encoder e : encoders) {
71              Class<?>[] typeArguments = TypeResolver.resolveArguments(e.getClass(), Encoder.class);
72  
73              if (typeArguments.length > 0 && typeArguments[0].isAssignableFrom(instanceType.getClass())) {
74                  instanceType = e.encode(instanceType);
75              }
76          }
77          return instanceType;
78      }

80      public Future write(Request request, Object data) throws IOException {
81          // Execute encoder
82          Object object = invokeEncoder(request.encoders(), data);
83          if (webSocket != null) {
84              webSocketWrite(request, object, data);
85          } else {
86              try {
87                  Response r = httpWrite(request, object, data).get(rootFuture.time, rootFuture.tu);
88                  String m = r.getResponseBody();
89                  if (!m.isEmpty()) {
90                      TransportsUtil.invokeFunction(request.decoders(),functions, String.class, m, MESSAGE.name(), request.functionResolver());
91                  }
92              } catch (TimeoutException t) {
93                  logger.trace("AHC Timeout", t);
94                  rootFuture.te = t;
95              } catch (Throwable t) {
96                  logger.error("", t);
97              }
98          }
99  
100         return rootFuture.done();
101     }

103     public void webSocketWrite(Request request, Object object, Object data) throws IOException {
104         if (InputStream.class.isAssignableFrom(object.getClass())) {
105             InputStream is = (InputStream) object;
106             ByteArrayOutputStream bs = new ByteArrayOutputStream();
107             //TODO: We need to stream directly, in AHC!
108             byte[] buffer = new byte[8192];
109             int n = 0;
110             while (-1 != (n = is.read(buffer))) {
111                 bs.write(buffer, 0, n);
112             }
113             webSocket.sendMessage(bs.toByteArray());
114         } else if (Reader.class.isAssignableFrom(object.getClass())) {
115             Reader is = (Reader) object;
116             StringWriter bs = new StringWriter();
117             //TODO: We need to stream directly, in AHC!
118             char[] chars = new char[8192];
119             int n = 0;
120             while (-1 != (n = is.read(chars))) {
121                 bs.write(chars, 0, n);
122             }
123             webSocket.sendTextMessage(bs.getBuffer().toString());
124         } else if (String.class.isAssignableFrom(object.getClass())) {
125             webSocket.sendTextMessage(object.toString());
126         } else if (byte[].class.isAssignableFrom(object.getClass())) {
127             webSocket.sendMessage((byte[]) object);
128         } else {
129             throw new IllegalStateException("No Encoder for " + data);
130         }
131     }

133     public ListenableFuture<Response> httpWrite(Request request, Object object, Object data) throws IOException {
134         // Only for Atmosphere
135         if (AtmosphereRequest.class.isAssignableFrom(request.getClass())) {
136             request.queryString().put("X-Atmosphere-Transport", Arrays.asList(new String[]{"polling"}));
137             request.queryString().remove("X-atmo-protocol");
138         }
139 
140         AsyncHttpClient.BoundRequestBuilder b = options.runtime().preparePost(request.uri())
141                 .setHeaders(request.headers())
142                 .setQueryParameters(DefaultSocket.decodeQueryString(request))
143                 .setMethod(Request.METHOD.POST.name());
144 
145         if (InputStream.class.isAssignableFrom(object.getClass())) {
146             //TODO: Allow reading the response.
147             return b.setBody((InputStream) object).execute();
148         } else if (Reader.class.isAssignableFrom(object.getClass())) {
149             return b.setBody(new ReaderInputStream((Reader) object)).execute();
150         } else if (String.class.isAssignableFrom(object.getClass())) {
151             return b.setBody((String) object) .execute();
152         } else if (byte[].class.isAssignableFrom(object.getClass())) {
153             return b.setBody((byte[]) object).execute();
154         } else {
155             throw new IllegalStateException("No Encoder for " + data);
156         }
157     }
