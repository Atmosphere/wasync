16  package org.atmosphere.wasync.impl;
17  
18  import com.ning.http.client.AsyncHttpClient;
19  import com.ning.http.client.AsyncHttpClientConfig;
20  import org.atmosphere.wasync.Client;
21  import org.atmosphere.wasync.Options;
22  import org.atmosphere.wasync.Socket;
23  
24  import static org.atmosphere.wasync.impl.AtmosphereRequest.AtmosphereRequestBuilder;
25  
26  public class AtmosphereClient implements Client<AtmosphereRequest.AtmosphereRequestBuilder> {
27      private AsyncHttpClient asyncHttpClient;
28  
29      public AtmosphereClient() {
30      }
31  
32      public Socket create() {
33          AsyncHttpClientConfig.Builder config = new AsyncHttpClientConfig.Builder();
34          config.setFollowRedirects(true).setRequestTimeoutInMs(-1).setUserAgent("wAsync/1.0");
35          asyncHttpClient = new AsyncHttpClient(config.build());
36          return getSocket(asyncHttpClient, new Options.OptionsBuilder().build());
37      }
38  
39      public Socket create(Options options) {
40          // TODO
41          AsyncHttpClientConfig.Builder config = new AsyncHttpClientConfig.Builder();
42          config.setFollowRedirects(true)
43                  .setRequestTimeoutInMs(-1)
44                  .setUserAgent("wAsync/1.0");
45  
46          asyncHttpClient = new AsyncHttpClient(config.build());
47          return getSocket(asyncHttpClient, options);
48      }
49  
50      @Override
51      public AtmosphereRequestBuilder newRequestBuilder(Class<AtmosphereRequest.AtmosphereRequestBuilder> clazz) {
52          AtmosphereRequestBuilder b = null;
53          try {
54              b = clazz.newInstance();
55          } catch (InstantiationException e) {
56              throw new RuntimeException(e);
57          } catch (IllegalAccessException e) {
58              throw new RuntimeException(e);
59          }
60          return AtmosphereRequestBuilder.class.cast(b.resolver(new DefaultFunctionResolver()));
61      }
62  
63      protected Socket getSocket(AsyncHttpClient asyncHttpClient, Options options) {
64          options.runtime(asyncHttpClient);
65          return new DefaultSocket(options);
66      }
67  
68      @Override
69      public AtmosphereRequestBuilder newRequestBuilder() {
70          AtmosphereRequestBuilder b = new AtmosphereRequestBuilder();
71          return AtmosphereRequestBuilder.class.cast(b.resolver(new DefaultFunctionResolver()));
72      }
73  }
