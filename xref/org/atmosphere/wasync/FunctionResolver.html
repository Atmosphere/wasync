16  package org.atmosphere.wasync;
17  
18  /**
19   * FunctionResolver are useful for mapping received message with a {@link Function}. By default, only the predefined
20   * function {@link org.atmosphere.wasync.Event} are automatically mapped to Function,
21   * <blockquote><pre>
22       Client client = ClientFactory.getDefault().newClient();
23  
24       RequestBuilder request = client.newRequestBuilder()
25               .method(Request.METHOD.GET)
26               .uri(targetUrl + "/suspend")
27               .decoder(new Decoder&lt;String, POJO&gt;() {
28                   &#64;Override
29                   public POJO decode(String s) {
30                       return new POJO(s);
31                   }
32               })
33               .transport(Request.TRANSPORT.WEBSOCKET);
34  
35       Socket socket = client.create();
36       socket.on(Function.EVENT_TYPE.message.name(), new Function&lt;POJO&gt;() {
37           &#64;Override
38           public void on(POJO t) {
39               response.set(t);
40           }
41       })
42   * </pre></blockquote>
43   * or when a {@link Decoder} share the same type as the
44   * defined Function. For example:
45   * <blockquote><pre>
46       Client client = ClientFactory.getDefault().newClient();
47  
48       RequestBuilder request = client.newRequestBuilder()
49               .method(Request.METHOD.GET)
50               .uri(targetUrl + "/suspend")
51               .decoder(new Decoder&lt;String, POJO&gt;() {
52                   &#64;Override
53                   public POJO decode(String s) {
54                       return new POJO(s);
55                   }
56               })
57               .transport(Request.TRANSPORT.WEBSOCKET);
58  
59       Socket socket = client.create();
60       socket.on(new Function&lt;POJO&gt;() {
61           &#64;Override
62           public void on(POJO t) {
63               response.set(t);
64           }
65       })
66   * </pre></blockquote>
67   * An application can define its own Function.EVENT_TYPE be writing the appropriate FunctionResolver.
68   *
69   * By default, the {@link #DEFAULT} is used.
70   *
71   * @author Jeanfrancois Arcand
72   */
73  public interface FunctionResolver {
74  
75      FunctionResolver DEFAULT = new FunctionResolver() {
76          /**
77           * {@inheritDoc}
78           */
79          @Override
80          public boolean resolve(String message, Object functionName, FunctionWrapper fn) {
81              return fn.functionName().isEmpty() || functionName.toString().equalsIgnoreCase(fn.functionName());
82          }
83      };
84  
85      /**
86       * Resolve the current message with
87       * @param message the original response's body received
88       * @param functionName the default function name taken from {@link org.atmosphere.wasync.Event} or from a custom FunctionResolver
89       * @param fn The current {@link FunctionWrapper}
90       * @return true if the {@link Function} can be invoked.
91       */
92      boolean resolve(String message, Object functionName, FunctionWrapper fn);
93  }
