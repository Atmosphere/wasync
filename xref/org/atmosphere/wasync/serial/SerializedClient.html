16  package org.atmosphere.wasync.serial;

18  import com.ning.http.client.AsyncHttpClient;
19  import org.atmosphere.wasync.Client;
20  import org.atmosphere.wasync.FunctionResolver;
21  import org.atmosphere.wasync.Socket;
22  import org.atmosphere.wasync.impl.AtmosphereRequest.AtmosphereRequestBuilder;
23  import org.atmosphere.wasync.impl.ClientUtil;
24  
25  /**
26   * {@code SerializedClient} is a {@link org.atmosphere.wasync.Client} that guarantees ordered message delivery, in-line with the
27   * {@link Socket#fire(Object)} invocation sequence.
28   * <p/>
29   * A sequence of fire calls over a {@code SerializedClient}'s socket (created through {@link SerializedClient#create()} :
30   * <blockquote><pre>
31   *     socket.fire("message1").fire("message2");
32   * </pre></blockquote>
33   * guarantees that {@code message1} arrives at the recipient-side before {@code message2}. By default, wAsync uses multiple underlying
34   * connections in delivering fire payloads. The {@code SerializedClient} guarantees that only one connection is used at any moment
35   * in time, while still providing an asynchronous fire interface to clients.
36   * <p/>
37   * {@code SerializedClient} instances can be configured by means of a {@link SerializedFireStage} in deciding on the exact
38   * staging semantics and the (non-functional) quality properties of a supporting stage. The default implementation provided is
39   * {@link DefaultSerializedFireStage}.
40   * <p/>
41   *
42   * @author Christian Bach
43   */
44  public class SerializedClient implements Client<SerializedOptions, SerializedOptionsBuilder, SerializedClient.SerializedRequestBuilder> {
45  
46      /**
47       * {@inheritDoc}
48       */
49      @Override
50      public Socket create(SerializedOptions options) {
51          AsyncHttpClient asyncHttpClient = options.runtime();
52          if (asyncHttpClient == null || asyncHttpClient.isClosed()) {
53              asyncHttpClient = ClientUtil.createDefaultAsyncHttpClient(options);
54              options.runtime(asyncHttpClient);
55          }
56          return new SerializedSocket(options);
57      }
58  
59      /**
60       * {@inheritDoc}
61       */
62      @Override
63      public Socket create() {
64          AsyncHttpClient asyncHttpClient = ClientUtil.createDefaultAsyncHttpClient(newOptionsBuilder().reconnect(true).build());
65  
66          return new SerializedSocket(
67                  new SerializedOptionsBuilder()
68                          .runtime(asyncHttpClient)
69                          .serializedFireStage(new DefaultSerializedFireStage())
70                          .build());
71      }
72  
73      /**
74       * {@inheritDoc}
75       */
76      @Override
77      public SerializedOptionsBuilder newOptionsBuilder() {
78          return new SerializedOptionsBuilder();
79      }
80  
81      /**
82       * {@inheritDoc}
83       */
84      @Override
85      public SerializedRequestBuilder newRequestBuilder() {
86          SerializedRequestBuilder b = new SerializedRequestBuilder();
87          return SerializedRequestBuilder.class.cast(b.resolver(FunctionResolver.DEFAULT));
88      }
89  
90      /**
91       * {@inheritDoc}
92       */
93      @Override
94      public SerializedRequestBuilder newRequestBuilder(Class<SerializedRequestBuilder> clazz) {
95          AtmosphereRequestBuilder b;
96          try {
97              b = clazz.newInstance();
98          } catch (InstantiationException e) {
99              throw new RuntimeException(e);
100         } catch (IllegalAccessException e) {
101             throw new RuntimeException(e);
102         }
103         return SerializedRequestBuilder.class.cast(b.resolver(FunctionResolver.DEFAULT));
104     }
105 
106     public static class SerializedRequestBuilder extends AtmosphereRequestBuilder {
107         public SerializedRequestBuilder() {
108             super();
109             enableProtocol(true);
110         }
111     }
112 }
