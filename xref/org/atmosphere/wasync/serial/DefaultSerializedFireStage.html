18  import com.google.common.util.concurrent.SettableFuture;
19  import com.ning.http.client.ListenableFuture;
20  import com.ning.http.client.Response;
21  
22  import java.util.ArrayList;
23  import java.util.concurrent.BlockingQueue;
24  import java.util.concurrent.ExecutorService;
25  import java.util.concurrent.Executors;
26  import java.util.concurrent.LinkedBlockingQueue;
27  
28  import org.atmosphere.wasync.Socket;

30  /**
31   * Default implementation of a {@link SerializedFireStage}.
32   * <p/>
33   * This implementation is based on an unbounded stage that enqueues the payload
34   * objects to be fired ({@link SerializedFireStage#enqueue(Object, SettableFuture)}) by 
35   * means of a {@link LinkedBlockingQueue}.
36   * <p/>
37   * Every instance of this class spans its dedicated stage thread, which sequentially 
38   * consumes payload objects off the stage queue.
39   * <p/>
40   * Binary payloads are aggregated up to a {@code maxBinaryPayloadAggregationSize}.
41   * <p/>
42   * @author Christian Bach
43   */
44  public class DefaultSerializedFireStage implements SerializedFireStage {
45  
46  	private volatile SerializedSocket socket;
47  	private final int maxBinaryMessagesAggregationSize;
48  	
49  	private final BlockingQueue<FirePayloadEntry> firePayloadsQueue;
50  	private final ExecutorService executorService; 
51  	private final Runnable fireTask;	 
52  	
53  	public DefaultSerializedFireStage() {
54  		this(4);		
55  	}
56  	
57  	public DefaultSerializedFireStage(int maxBinaryPayloadAggregationSize) {
58  		this.maxBinaryMessagesAggregationSize = maxBinaryPayloadAggregationSize;
59  		firePayloadsQueue = new LinkedBlockingQueue<FirePayloadEntry>();
60  		executorService = Executors.newSingleThreadExecutor();
61  		fireTask = createFireTask();
62  		executorService.execute(fireTask);
63  	}
64  	
65  	@Override
66  	public void setSocket(SerializedSocket socket) {
67  		this.socket = socket;
68  	}
69  	
70  	@Override
71  	public void enqueue(Object firePayload, SettableFuture<Response> originalFuture) {
72  		firePayloadsQueue.add(new FirePayloadEntry(firePayload, originalFuture));
73  	}
74  	
75  	private Runnable createFireTask() {
76  		return new Runnable() {
77  			public void run() {
78  				ArrayList<FirePayloadEntry> aggregatedByteArrayPayloads = new ArrayList<FirePayloadEntry>(maxBinaryMessagesAggregationSize);				
79  				try {
80  					while(!Thread.currentThread().isInterrupted()) {
81  						FirePayloadEntry payloadEntry = firePayloadsQueue.take();						
82  						int aggregationCount = 0;
83  						
84  						for(; aggregationCount < maxBinaryMessagesAggregationSize; aggregationCount++) {
85  							if (byte[].class.isAssignableFrom(payloadEntry.getFirePayload().getClass())) {
86  								aggregatedByteArrayPayloads.add(payloadEntry);
87  								payloadEntry = firePayloadsQueue.poll();
88  								if (payloadEntry == null) {
89  									aggregationCount = 0;
90  									break;
91  								}
92  							} else {
93  								if (!aggregatedByteArrayPayloads.isEmpty()) {
94  									fireSynchronously(aggregatedByteArrayPayloads);
95  									aggregatedByteArrayPayloads.clear();
96  								} 
97  								fireSynchronously(payloadEntry);
98  								aggregationCount = 0;
99  								break;
100 							}
101 						}
102 						
103 						if(!aggregatedByteArrayPayloads.isEmpty()) {
104 							fireSynchronously(aggregatedByteArrayPayloads);
105 							aggregatedByteArrayPayloads.clear();					
106 						}
107 							
108 					}
109 				}catch (InterruptedException consumed) {
110 					// allow thread to exit
111 				}
112 			}
113 		};
114 	}
115 	
116 	private void fireSynchronously(ArrayList<FirePayloadEntry> aggregatedByteArrayPayloads) {
117 		ListenableFuture<Response> future;
118 		int aggregatedSize = 0;
119 		for (FirePayloadEntry entry : aggregatedByteArrayPayloads) {
120 			aggregatedSize += ((byte[])entry.getFirePayload()).length;
121 		}
122 		byte[] aggregatedByteArray = new byte[aggregatedSize];
123 		int destPos = 0;
124 		for (FirePayloadEntry entry : aggregatedByteArrayPayloads) {
125 			byte[] payload = (byte[])entry.getFirePayload();
126 			System.arraycopy(
127 					payload, 0, 
128 					aggregatedByteArray, destPos, 
129 					payload.length);
130 			destPos += payload.length;
131 		}
132 		
133 	
134 		Response response = null;
135 		try {
136 			future = socket.directWrite(aggregatedByteArray);
137 			response = future.get();
138 		} catch (Exception e) {				
139 			for (FirePayloadEntry entry : aggregatedByteArrayPayloads) {
140 				entry.getOriginalFuture().setException(e);
141 				entry.getOriginalFuture().cancel(true);
142 			}
143 		} finally {
144 			for (FirePayloadEntry entry : aggregatedByteArrayPayloads) {
145 				entry.getOriginalFuture().set(response);
146 			}
147 		}								
148 	}
149 	
150 	public void fireSynchronously(FirePayloadEntry firePayloadEntry) {
151 		ListenableFuture<Response> future;
152 		Response response = null;
153 		try {
154 			future = socket.directWrite(firePayloadEntry.firePayload);
155 			response = future.get();
156 		} catch (Exception e) {				
157 			firePayloadEntry.getOriginalFuture().setException(e);
158 			firePayloadEntry.getOriginalFuture().cancel(true);
159 		} finally {			
160 			firePayloadEntry.getOriginalFuture().set(response);
161 		}
162 	}
163 	
164 	@Override
165 	public void shutdown() {
166 		executorService.shutdownNow();
167 		for (FirePayloadEntry entry  : firePayloadsQueue) {
168 			entry.getOriginalFuture().cancel(true);
169 		}
170 	} 
171 	
172 	private class FirePayloadEntry {
173 		
174 		private Object firePayload;
175 		private SettableFuture<Response> originalFuture;
176 		
177 		public FirePayloadEntry(Object firePayload, SettableFuture<Response> originalFuture) {
178 			this.firePayload = firePayload;
179 			this.originalFuture = originalFuture;
180 		}
181 		
182 		public Object getFirePayload() { 
183 			return firePayload;
184 		}
185 		
186 		public SettableFuture<Response> getOriginalFuture() { 
187 			return originalFuture;
188 		}
189 		
190 	}
