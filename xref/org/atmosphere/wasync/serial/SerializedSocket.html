35  /**
36   * {@code SerializedSocket} is a {@link Socket} implementation that guarantees ordered message delivery of
37   * {@link Socket#fire(Object)} calls, by serializing fire calls over a {@link SerializedFireStage}.
38   * <p/>
39   * {@code SerializedSocket} guarantees to use only one underlying connection at any moment in time, while still
40   * providing an asynchronous fire interface to clients.
41   * <p/>
42   *
43   * @author Christian Bach
44   */
45  public class SerializedSocket extends DefaultSocket {
46  
47      private final static Logger logger = LoggerFactory.getLogger(SerializedSocket.class);
48  
49      private SerializedFireStage serializedFireStage;
50      private AsyncHttpClient asyncHttpClient;
51      
52      public SerializedSocket(SerializedOptions options) {
53          super(options);
54          if (options.runtime() == null || options.runtime().isClosed()) {
55              asyncHttpClient = ClientUtil.createDefaultAsyncHttpClient(options);
56              options.runtime(asyncHttpClient);
57          }
58          this.serializedFireStage = options.serializedFireStage();
59          this.serializedFireStage.setSocket(this);
60      }
61  
62      /**
63       * {@inheritDoc}
64       */
65      @Override
66      public SocketRuntime createRuntime(DefaultFuture future, Options options, List<FunctionWrapper> functions) {
67          return new SerialSocketRuntime(transportInUse, options, new DefaultFuture(this), this, functions);
68      }
69  
70      public SerializedFireStage getSerializedFireStage() {
71          return serializedFireStage;
72      }
73  
74      public ListenableFuture<Response> directWrite(Object encodedPayload) throws IOException {
75          return socketRuntime.httpWrite(request, encodedPayload, encodedPayload);
76      }
77      
78      /**
79       * {@inheritDoc}
80       */
81      @Override
82      public void close() {
83      	serializedFireStage.shutdown();
84      	if (asyncHttpClient != null) {
85      		asyncHttpClient.close();
86      	}
87      	super.close();
88      }
89      
90  }
