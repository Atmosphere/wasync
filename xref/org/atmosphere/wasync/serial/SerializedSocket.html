29  /**
30   * {@code SerializedSocket} is a {@link Socket} implementation that guarantees ordered message delivery of
31   * {@link Socket#fire(Object)} calls, by serializing fire calls over a {@link SerializedFireStage}.
32   * <p/>
33   * {@code SerializedSocket} guarantees to use only one underlying connection at any moment in time, while still
34   * providing an asynchronous fire interface to clients.
35   * <p/>
36   *
37   * @author Christian Bach
38   * @author Christian Bach
39   */
40  public class SerializedSocket extends DefaultSocket {
41  
42      private final static Logger logger = LoggerFactory.getLogger(SerializedSocket.class);
43  
44      private SerializedFireStage serializedFireStage;
45  
46      public SerializedSocket(SerializedOptions options) {
47          super(options);
48          this.serializedFireStage = options.serializedFireStage();
49          this.serializedFireStage.setSocket(this);
50      }
51  
52      @Override
53      protected SocketRuntime createSocket() {
54          return new SerialSocketRuntime(options, new DefaultFuture(this), this, functions);
55      }
56  
57      public SerializedFireStage getSerializedFireStage() {
58          return serializedFireStage;
59      }
60  
61      public ListenableFuture<Response> directWrite(Object encodedPayload) throws IOException {
62          return socketRuntime.httpWrite(request, encodedPayload, encodedPayload);
63      }
64  }
