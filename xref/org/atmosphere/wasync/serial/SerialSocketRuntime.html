41  /**
42   * Serial extension for the {@link SocketRuntime}
43   *
44   * @author Jeanfrancois Arcand
45   *
46   */
47  public class SerialSocketRuntime extends SocketRuntime {
48  
49      private final static Logger logger = LoggerFactory.getLogger(SerialSocketRuntime.class);
50      private final SerializedSocket serializedSocket;
51  
52      public SerialSocketRuntime(Transport transport, Options options, DefaultFuture rootFuture, SerializedSocket serializedSocket,  List<FunctionWrapper> functions) {
53          super(transport, options, rootFuture, functions);
54          this.serializedSocket = serializedSocket;
55      }
56  
57      @Override
58      public Future write(Request request, Object data) throws IOException {
59  
60          if (WebSocketTransport.class.isAssignableFrom(transport.getClass())) {
61              Object object = invokeEncoder(request.encoders(), data);
62              webSocketWrite(request, object, data);
63          } else {
64              // Execute encoder
65              Object encodedPayload = invokeEncoder(request.encoders(), data);
66              if (!(InputStream.class.isAssignableFrom(encodedPayload.getClass())
67                              || Reader.class.isAssignableFrom(encodedPayload.getClass())
68                              || String.class.isAssignableFrom(encodedPayload.getClass())
69                              || byte[].class.isAssignableFrom(encodedPayload.getClass())
70                      )) {
71                  throw new IllegalStateException("No Encoder for " + data);
72              }
73  
74              if (serializedSocket.getSerializedFireStage() != null) {
75                  final SettableFuture<Response> future = SettableFuture.create();
76                  serializedSocket.getSerializedFireStage().enqueue(encodedPayload, future);
77                  return new Future() {
78  
79                      @Override
80                      public Future fire(Object data) throws IOException {
81                          return serializedSocket.fire(data);
82                      }
83  
84                      @Override
85                      public Future done() {
86                          ListenableFuture.class.cast(future).done();
87                          return this;
88                      }
89  
90                      @Override
91                      public boolean cancel(boolean mayInterruptIfRunning) {
92                          return future.cancel(mayInterruptIfRunning);
93                      }
94  
95                      @Override
96                      public boolean isCancelled() {
97                          return future.isCancelled();
98                      }
99  
100                     @Override
101                     public boolean isDone() {
102                         return future.isDone();
103                     }
104 
105                     @Override
106                     public Socket get() throws InterruptedException, ExecutionException {
107                         future.get();
108                         return serializedSocket;
109                     }
110 
111                     @Override
112                     public Socket get(long timeout, TimeUnit unit) throws InterruptedException, ExecutionException, TimeoutException {
113                         future.get(timeout, unit);
114                         return serializedSocket;
115                     }
116                 };
117             } else {
118                 final ListenableFuture<Response> future = serializedSocket.directWrite(encodedPayload);
119                 return new Future() {
120                     @Override
121                     public Future fire(Object data) throws IOException {
122                         return serializedSocket.fire(data);
123                     }
124 
125                     @Override
126                     public Future done() {
127                         future.done();
128                         return this;
129                     }
130 
131                     @Override
132                     public boolean cancel(boolean mayInterruptIfRunning) {
133                         return future.cancel(mayInterruptIfRunning);
134                     }
135 
136                     @Override
137                     public boolean isCancelled() {
138                         return future.isCancelled();
139                     }
140 
141                     @Override
142                     public boolean isDone() {
143                         return future.isDone();
144                     }
145 
146                     @Override
147                     public Socket get() throws InterruptedException, ExecutionException {
148                         future.get();
149                         return serializedSocket;
150                     }
151 
152                     @Override
153                     public Socket get(long timeout, TimeUnit unit) throws InterruptedException, ExecutionException, TimeoutException {
154                         future.get(timeout, unit);
155                         return serializedSocket;
156                     }
157                 };
158             }
159         }
160         return rootFuture;
161     }
