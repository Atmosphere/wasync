16  package org.atmosphere.wasync;
17  
18  import com.ning.http.client.AsyncHttpClient;
19  
20  /**
21   * Configure the underlying WebSocket/HTTP client.
22   *
23   * @author Jeanfrancois Arcand
24   */
25  public interface Options {
26      /**
27       * The used {@link Transport}
28       * @return {@link Transport}
29       */
30      public Transport transport() ;
31  
32      /**
33       * Reconnect after a network failure or when the server close the connection.
34       * @return reconnect
35       */
36      public boolean reconnect();
37  
38      /**
39       * The delay, in second, before reconnecting.
40       * @return The delay, in second, before reconnecting.
41       */
42      public int reconnectInSeconds();
43  
44      /**
45       * When using long-polling and the {@link Request}, the delay before considering the long-polling connection has been fully processed by the server. If you use
46       * the {@link org.atmosphere.wasync.impl.AtmosphereClient}, the server will send some handshake so this value is not needed.
47       * @return the delay before considering the long-polling connection has been fully processed
48       */
49      public long waitBeforeUnlocking();
50  
51      /**
52       * The {@link AsyncHttpClient} used to communicate with server.
53       * @return {@link AsyncHttpClient} used to communicate with server.
54       */
55      public AsyncHttpClient runtime();
56  
57      /**
58       * Set the {@link AsyncHttpClient}.
59       * @param client the {@link AsyncHttpClient}
60       */
61      public void runtime(AsyncHttpClient client);
62  
63      /**
64       * Return true is the {@link AsyncHttpClient} is shared between {@link Socket}. Default is false. You need to invoke {@link #runtime(com.ning.http.client.AsyncHttpClient)} to make
65       * it shared.
66       * @return true is the {@link AsyncHttpClient}
67       */
68      public boolean runtimeShared();
69  
70      /**
71       * The time, in seconds to wait before closing the connection.
72       * @return The time, in seconds to wait before closing the connection.
73       */
74      public int requestTimeoutInSeconds() ;
75  }
