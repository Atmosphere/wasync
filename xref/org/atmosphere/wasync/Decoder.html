18  /**
19   * A decoder can be used to 'decode' the response's body. Decoder can be chained amongst themselves in order to create
20   * plain Java Object. Decoder's can be used to decode the response body and match them with {@link Function}'s implementation.
21   * This library will try to match the decoded Object to its associated Function. For example, the Decoder's type will
22   * be mapped, by the {@link FunctionResolver} to the Function of the same type if no function message has been defined:
23   *
24   * <blockquote><pre>
25  
26     Decoder&lt;String, POJO&gt; d = new Decoder&lt;String, POJO&gt;() {
27               &#64;Override
28               public POJO decode(Event e, String s) {
29                   return new POJO(s);
30               }
31           }
32  
33     Function&lt;POJO&gt; f = new Function&lt;POJO&gt;() {
34               &#64;Override
35               public void on(Event e, POJO t) {
36  
37               }
38          }
39  
40   * </pre></blockquote>
41   * @param <U>
42   * @param <T>
43   * @author Jeanfrancois Arcand
44   */
45  public interface Decoder<U extends Object, T> {
46      /**
47       * Decode the specified object of type U into object of type T
48       *
49       * @param e Event
50       * @param s a object of type U
51       * @return a new object of type T
52       */
53      T decode(Event e, U s);
54  
55  }
