16  package org.atmosphere.wasync;
17  
18  /**
19   * A decoder can be used to 'decode' the response's body. Decoder's can be used to decode the response body and match them with {@link Function}'s implementation.
20   * This library will try to match the decoded Object to its associated Function. For example, the Decoder's type will
21   * be mapped, by the {@link FunctionResolver} to the Function of the same type if no function message has been defined:
22   *
23   * <blockquote><pre>
24  
25     Decoder&lt;String, POJO&gt; d = new Decoder&lt;String, POJO&gt;() {
26               &#64;Override
27               public POJO decode(Event e, String s) {
28                   return new POJO(s);
29               }
30           }
31   * </pre></blockquote>
32   * Important: Decoder cannot be chained, e.g the result of a Decoder will never be passed to another Decoder, like {@link Encoder}
33   * @param <U>
34   * @param <T>
35   * @author Jeanfrancois Arcand
36   */
37  public interface Decoder<U, T> {
38      /**
39       * Decode the specified object of type U into object of type T.
40       *
41       * @param e an {@link Event} type. This can be used to differentiate event received.
42       * @param s a object of type U
43       * @return a new object of type T
44       */
45      T decode(Event e, U s);
46  
47      /**
48       * A Decoder may return an instance of a Decoded object to prevent some messages from being delivered to a {@link Function},
49       * by returning a {@link Decoded#ABORT} instance. Returning an instance of Decoded with the action set to {@link Decoded.ACTION#CONTINUE}
50       * will allow dispatching messages to {@link Function}
51       * @param <T>
52       */
53      public final static class Decoded<T> {
54          /**
55           * Use this object to prevent the delivering of messages to {@link Function}
56           */
57          public final static Decoded ABORT = new Decoded(null, ACTION.ABORT);
58  
59          public enum ACTION {
60              /**
61               * Continue the decoding/dispatching process.
62               */
63              CONTINUE,
64              /**
65               * Do not dispatch the message to {@link Function}
66               */
67              ABORT}
68  
69          private final T decodedMessage;
70          private final ACTION action;
71  
72          /**
73           * Create a decoded object
74           * @param decodedMessage the decoded message
75           * @param action the action
76           */
77          public Decoded(T decodedMessage, ACTION action) {
78              this.decodedMessage = decodedMessage;
79              this.action = action;
80          }
81  
82          /**
83           * Create a decoded object with action set to {@link ACTION#CONTINUE}
84           * @param decodedMessage
85           */
86          public Decoded(T decodedMessage) {
87              this.decodedMessage = decodedMessage;
88              this.action = ACTION.CONTINUE;
89          }
90  
91          /**
92           * Return the decoded message.
93           * @return the decoded message.
94           */
95          public T decoded() {
96              return decodedMessage;
97          }
98  
99          /**
100          * Return the action.
101          * @return the action.
102          */
103         public ACTION action() {
104             return action;
105         }
106     }
107 
108 }
