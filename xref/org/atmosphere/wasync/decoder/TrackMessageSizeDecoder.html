16  package org.atmosphere.wasync.decoder;
17  
18  import org.atmosphere.wasync.Event;
19  import org.atmosphere.wasync.ReplayDecoder;
20  import org.atmosphere.wasync.Transport;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import java.util.ArrayList;
25  import java.util.List;
26  
27  public class TrackMessageSizeDecoder implements ReplayDecoder {
28  
29      private final Logger logger = LoggerFactory.getLogger(TrackMessageSizeDecoder.class);
30  
31      private final String delimiter;
32      private final StringBuffer messagesBuffer = new StringBuffer();
33  
34      public TrackMessageSizeDecoder() {
35          this.delimiter = "|";
36      }
37  
38      public TrackMessageSizeDecoder(String delimiter) {
39          this.delimiter = delimiter;
40      }
41  
42      @Override
43      public List<String> decode(Event type, String message) {
44          if (type.equals(Event.MESSAGE)) {
45              ArrayList<String> messages = new ArrayList<String>();
46  
47              int messageLength = -1;
48              int messageStartIndex = 0;
49              int delimiterIndex = -1;
50              String singleMessage = null;
51              while ((delimiterIndex = message.indexOf(delimiter, messageStartIndex)) >= 0) {
52                  if (delimiterIndex == messageStartIndex) {
53                      messageStartIndex = delimiterIndex + 1;
54                      continue;
55                  }
56                  try {
57                      messageLength = Integer.valueOf(message.substring(messageStartIndex, delimiterIndex));
58                      if (messageLength <= 0) {
59                          throw new Exception();
60                      }
61                  } catch (Exception e) {
62                      logger.error("", e);
63                      //discard whole message
64                      messagesBuffer.setLength(0);
65                      throw new Error("Message format is not as expected for tracking message size"); //this error causes invocation of onThrowable of AsyncHandler if not caught in between
66                  }
67  
68                  messageStartIndex = delimiterIndex < (message.length() - 1) ? delimiterIndex + 1 : message.length();
69                  int lenghtOfRemainingMessage = (message.length() - messageStartIndex);
70                  singleMessage = message.substring(messageStartIndex, messageLength <= lenghtOfRemainingMessage
71                          ? messageStartIndex + messageLength : messageStartIndex + lenghtOfRemainingMessage);
72  
73                  delimiterIndex = message.indexOf(delimiter, messageStartIndex + messageLength);
74                  if (delimiterIndex >= 0) {
75                      messageStartIndex = delimiterIndex < (message.length() - 1) ? delimiterIndex + 1 : (message.length() - 1);
76                  }
77  
78                  if (singleMessage.length() == messageLength) {
79                      messages.add(singleMessage);
80                      if (delimiterIndex < 0) {
81                          messagesBuffer.setLength(0);
82                      }
83                  } else {
84                      messagesBuffer.setLength(0);
85                      messagesBuffer.append(messageLength).append(delimiter).append(singleMessage);
86                  }
87              }
88              return messages;
89          } else {
90              ArrayList<String> l = new ArrayList<String>();
91              l.add(message);
92              return l;
93          }
94      }
95  }
