16  package org.atmosphere.wasync.decoder;
17  
18  import org.atmosphere.wasync.Event;
19  import org.atmosphere.wasync.ReplayDecoder;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  import java.util.Collections;
24  import java.util.LinkedList;
25  import java.util.List;
26  import java.util.concurrent.atomic.AtomicBoolean;
27  
28  public class TrackMessageSizeDecoder implements ReplayDecoder {
29  
30      private final Logger logger = LoggerFactory.getLogger(TrackMessageSizeDecoder.class);
31  
32      private final String delimiter;
33      private final StringBuffer messagesBuffer = new StringBuffer();
34      private final AtomicBoolean skipFirstMessage = new AtomicBoolean();
35      private final List<String> empty = Collections.<String>emptyList();
36  
37      public TrackMessageSizeDecoder() {
38          this.delimiter = "\\|";
39      }
40  
41      public TrackMessageSizeDecoder(boolean protocolEnabled) {
42          this.delimiter = "\\|";
43          skipFirstMessage.set(protocolEnabled);
44      }
45  
46      public TrackMessageSizeDecoder(String delimiter, boolean protocolEnabled) {
47          this.delimiter = delimiter;
48          skipFirstMessage.set(protocolEnabled);
49      }
50  
51      @Override
52      public List<String> decode(Event type, String message) {
53          if (type.equals(Event.MESSAGE)) {
54  
55              if (skipFirstMessage.getAndSet(false)) return empty;
56              LinkedList<String> messages = new LinkedList<String>();
57  
58              message = messagesBuffer.append(message).toString().replace(delimiter, "__");
59              messagesBuffer.setLength(0);
60  
61              if (message.indexOf("__") != -1) {
62                  String[] tokens = message.split("__");
63  
64                  // Skip first.
65                  int pos = 1;
66                  int length = Integer.valueOf(tokens[0]);
67                  String m;
68                  while (pos < tokens.length) {
69                      m = tokens[pos];
70                      if (m.length() >= length) {
71                          messages.addLast(m.substring(0, length));
72                          String t = m.substring(length);
73                          if (!t.isEmpty()) {
74                              length = Integer.valueOf(t);
75                          }
76                      } else {
77                          if (pos != 1) {
78                              messagesBuffer.append(length).append(delimiter).append(m);
79                          } else {
80                              messagesBuffer.append(message);
81                          }
82                          break;
83                      }
84                      pos++;
85                  }
86              }
87  
88              return messages;
89          } else {
90              return empty;
91          }
92      }
93  }
