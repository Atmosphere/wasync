16  package org.atmosphere.wasync.decoder;
17  
18  import org.atmosphere.wasync.Decoder;
19  import org.atmosphere.wasync.Decoder.Decoded;
20  import org.atmosphere.wasync.Event;
21  
22  /**
23   * Trim white space protocol sent by the Atmosphere's org.atmosphere.interceptor.PaddingAtmosphereInterceptor,
24   * discard heartbeat message from the server.
25   *
26   * @Author Jean-Francois Arcand
27   */
28  public class PaddingAndHeartbeatDecoder implements Decoder<String, Decoded<String>> {
29  
30      private final int paddingSize;
31      private final String heartbeatChar;
32  
33      public PaddingAndHeartbeatDecoder(){
34          this(4098, "X");
35      }
36  
37      public PaddingAndHeartbeatDecoder(int paddingSize, String heartbeatChar) {
38          this.paddingSize = paddingSize;
39          this.heartbeatChar = heartbeatChar;
40      }
41  
42      @Override
43      public Decoded<String>  decode(Event type, String message) {
44          if (type.equals(Event.MESSAGE)) {
45  
46              if (message.equalsIgnoreCase(heartbeatChar)) {
47                  return new Decoded<String>(message, Decoded.ACTION.ABORT);
48              }
49  
50              message = ltrim(message);
51              if (message == null) {
52                  return new Decoded<String>(message, Decoded.ACTION.ABORT);
53              }
54          }
55          return new Decoded<String>(message);
56      }
57  
58      private String ltrim(String s) {
59  
60          int i = 0;
61          while (i < s.length() && Character.isWhitespace(s.charAt(i))) {
62              i++;
63          }
64  
65          return i == paddingSize ? s.trim().length() == 0 ? null : s.substring(i) : s;
66      }
67  
68  }
