16  package org.atmosphere.wasync;
17  
18  /**
19   * A tupple which contains a {@link Function} and its associated functionName. The values are passed from
20   * <blockquote><pre>
21   *
22   *     socket.on("message, new Function&lt;String&gt;() {
23   *         ....
24   *     }
25   * </pre></blockquote>
26   * This class is only used by {@link Transport} implementation.
27   *
28   * @author Jeanfrancois Arcand
29   */
30  public class FunctionWrapper {
31  
32      private final String functionName;
33      private final Function<?> function;
34  
35      public FunctionWrapper(String functionName, Function<?> function) {
36          this.functionName = functionName;
37          this.function = function;
38      }
39  
40      public Function<?> function(){
41          return function;
42      }
43  
44      public String functionName() {
45          return functionName;
46      }
47  }
