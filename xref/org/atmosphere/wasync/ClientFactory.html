16  package org.atmosphere.wasync;
17  
18  import org.atmosphere.wasync.impl.DefaultClient;
19  
20  /**
21   * Create a {@link Client}. By default, the {@link DefaultClient} will be returned. To override the {@link DefaultClient},
22   * just specify the provider using the -Dwasync.client property.
23   *
24   * @author Jeanfrancois Arcand
25   */
26  public class ClientFactory {
27  
28      private final static ClientFactory factory = new ClientFactory();
29      private final String clientClassName;
30  
31      public ClientFactory() {
32          clientClassName = System.getProperty("wasync.client");
33      }
34  
35      /**
36       * Return the default Factory.
37       *
38       * @return this
39       */
40      public final static ClientFactory getDefault() {
41          return factory;
42      }
43  
44      /**
45       * Return a new {@link Client} instance
46       *
47       * @return a new {@link Client} instance
48       */
49      public Client newClient() {
50          if (clientClassName == null) {
51              return new DefaultClient();
52          } else {
53              try {
54                  return (Client) Thread.currentThread().getContextClassLoader().loadClass(clientClassName).newInstance();
55              } catch (Exception e) {
56                  throw new RuntimeException(e);
57              }
58          }
59      }
60  }
