29  public class TransportsUtil {

31      static void invokeFunction(List<Decoder<? extends Object, ?>> decoders,
32                                 List<FunctionWrapper> functions,
33                                 Class<?> implementedType,
34                                 Object instanceType,
35                                 String functionName,
36                                 FunctionResolver resolver) {
37          invokeFunction(Transport.EVENT_TYPE.MESSAGE, decoders, functions, implementedType, instanceType, functionName, resolver);
38      }

40      static void invokeFunction(Transport.EVENT_TYPE e,
41                                 List<Decoder<? extends Object, ?>> decoders,
42                                 List<FunctionWrapper> functions,
43                                 Class<?> implementedType,
44                                 Object instanceType,
45                                 String functionName,
46                                 FunctionResolver resolver) {
47  
48          String originalMessage = instanceType.toString();
49          for (FunctionWrapper wrapper : functions) {
50              Function f = wrapper.function();
51              Class<?>[] typeArguments = TypeResolver.resolveArguments(f.getClass(), Function.class);
52  
53              if (typeArguments.length > 0 && instanceType != null) {
54                  instanceType = matchDecoder(e, instanceType, decoders);
55                  if (instanceType != null) {
56                      implementedType = instanceType.getClass();
57                  }
58              }
59  
60              if (typeArguments.length > 0 && typeArguments[0].isAssignableFrom(implementedType)) {
61                  if (resolver.resolve(originalMessage, functionName, wrapper)) {
62                      f.on(instanceType);
63                  }
64              }
65          }
66      }
67  
68      static Object matchDecoder(Transport.EVENT_TYPE e, Object instanceType, List<Decoder<? extends Object, ?>> decoders) {
69          for (Decoder d : decoders) {
70              Class<?>[] typeArguments = TypeResolver.resolveArguments(d.getClass(), Decoder.class);
71              if (instanceType != null && typeArguments.length > 0 && typeArguments[0].equals(instanceType.getClass())) {
72                  boolean replay = ReplayDecoder.class.isAssignableFrom(d.getClass());
73  
74                  instanceType = d.decode(e, instanceType);
75  
76                  if (replay) {
77                      List<?> l = List.class.cast(instanceType);
78                      List<Decoder<? extends Object, ?>> nd = new ArrayList<Decoder<? extends Object, ?>>();
79                      boolean add = false;
80                      for (Decoder d2: decoders) {
81                          if (d2.equals(d)){
82                              add = true;
83                              continue;
84                          }
85  
86                          if (add) nd.add(d2);
87                      }
88  
89                      for (Object m : l) {
90                          return matchDecoder(e, m, nd);
91                      }
92                  }
93              }
94          }
95          return instanceType;
96      }
