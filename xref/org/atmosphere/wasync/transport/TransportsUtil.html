31  public class TransportsUtil {

33      private final static Logger logger = LoggerFactory.getLogger(TransportsUtil.class);

35      public static boolean invokeFunction(List<Decoder<? extends Object, ?>> decoders,
36                                 List<FunctionWrapper> functions,
37                                 Class<?> implementedType,
38                                 Object instanceType,
39                                 String functionName,
40                                 FunctionResolver resolver) {
41          return invokeFunction(Event.MESSAGE, decoders, functions, implementedType, instanceType, functionName, resolver);
42      }

44      public static boolean invokeFunction(Event e,
45                                 List<Decoder<? extends Object, ?>> decoders,
46                                 List<FunctionWrapper> functions,
47                                 Class<?> implementedType,
48                                 Object instanceType,
49                                 String functionName,
50                                 FunctionResolver resolver) {
51          boolean hasMatch = false;
52          String originalMessage = instanceType == null? "" : instanceType.toString();
53          for (FunctionWrapper wrapper : functions) {
54              Function f = wrapper.function();
55              Class<?>[] typeArguments = TypeResolver.resolveArguments(f.getClass(), Function.class);
56  
57              if (typeArguments.length > 0 && instanceType != null) {
58                  instanceType = matchDecoder(e, instanceType, decoders);
59                  if (instanceType != null) {
60                      implementedType = instanceType.getClass();
61                  }
62              }
63  
64              if (instanceType != null && typeArguments.length > 0 && typeArguments[0].isAssignableFrom(implementedType)
65                      // Don't invoke function with Event message.
66                      && !e.name().equalsIgnoreCase(instanceType.toString())) {
67                  if (resolver.resolve(originalMessage, functionName, wrapper)) {
68                      hasMatch = true;
69                      logger.trace("{} .on {}", functionName, instanceType);
70                      f.on(instanceType);
71                  }
72              }
73          }
74  
75          if (!hasMatch && !e.equals(Event.MESSAGE)) {
76              // Since we have no match, most probably because a decoder isn't matching a function or the Event's type, try
77              // to match Event type directly with a String.
78              // This can happens if a decoder is not behaving properly.
79              // instanceType != null because a ReplayDecoder may have interrupted
80              for (FunctionWrapper wrapper : functions) {
81                  Function f = wrapper.function();
82                  if (wrapper.functionName().equalsIgnoreCase(functionName)) {
83                      hasMatch = true;
84                      logger.trace("{} .on {}", functionName, instanceType);
85                      f.on(originalMessage);
86                  }
87              }
88          }
89  
90          return hasMatch;
91      }

93      public static Object matchDecoder(Event e, Object instanceType, List<Decoder<? extends Object, ?>> decoders) {
94          for (Decoder d : decoders) {
95              Class<?>[] typeArguments = TypeResolver.resolveArguments(d.getClass(), Decoder.class);
96              if (instanceType != null && typeArguments.length > 0 && typeArguments[0].isAssignableFrom(instanceType.getClass())) {
97                  boolean replay = ReplayDecoder.class.isAssignableFrom(d.getClass());
98  
99                  logger.trace("{} is trying to decode {}", d, instanceType);
100                 instanceType = d.decode(e, instanceType);
101 
102                 if (instanceType != null) {
103                     logger.trace("Decoder {} match {}", d, instanceType);
104                 }
105 
106                 if (replay) {
107                     List<?> l = List.class.cast(instanceType);
108                     List<Decoder<? extends Object, ?>> nd = new ArrayList<Decoder<? extends Object, ?>>();
109                     boolean add = false;
110                     for (Decoder d2: decoders) {
111                         if (d2.equals(d)){
112                             add = true;
113                             continue;
114                         }
115 
116                         if (add) nd.add(d2);
117                     }
118 
119                     for (Object m : l) {
120                         return matchDecoder(e, m, nd);
121                     }
122                 }
123             }
124         }
125         return instanceType;
126     }
