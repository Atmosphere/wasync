16  package org.atmosphere.wasync.transport;
17  
18  import com.ning.http.client.AsyncHandler;
19  import com.ning.http.client.HttpResponseBodyPart;
20  import com.ning.http.client.HttpResponseHeaders;
21  import com.ning.http.client.HttpResponseStatus;
22  import com.ning.http.client.RequestBuilder;
23  import org.atmosphere.wasync.FunctionWrapper;
24  import org.atmosphere.wasync.Options;
25  import org.atmosphere.wasync.Request;
26  import org.atmosphere.wasync.Socket;
27  import org.atmosphere.wasync.util.Utils;
28  
29  import java.util.List;
30  import java.util.concurrent.atomic.AtomicBoolean;
31  
32  import static org.atmosphere.wasync.Event.MESSAGE;
33  
34  /**
35   * Long-Polling {@link org.atmosphere.wasync.Transport} implementation
36   *
37   * @author Jeanfrancois Arcand
38   */
39  public class LongPollingTransport extends StreamTransport {
40  
41      /**
42       * When Atmosphere Protocol is used, we must not invoke any Function until the protocol has been processed.
43       */
44      private final AtomicBoolean handshakeOccured = new AtomicBoolean(true);
45      protected boolean protocolReceived = false;
46      private int count = 0;
47  
48      public LongPollingTransport(RequestBuilder requestBuilder, Options options, Request request, List<FunctionWrapper> functions) {
49          super(requestBuilder, options, request, functions);
50          List<String> protocol = request.queryString().get("X-atmo-protocol");
51          List<String> transport = request.queryString().get("X-Atmosphere-Transport");
52          if (protocol != null && transport != null
53                  && protocol.get(0).equals("true")
54                  && transport.get(0).equals("long-polling")) {
55              handshakeOccured.set(false);
56          }
57      }
58  
59      /**
60       * {@inheritDoc}
61       */
62      @Override
63      public STATE onHeadersReceived(HttpResponseHeaders headers) throws Exception {
64          if (handshakeOccured.get()) {
65              return super.onHeadersReceived(headers);
66          }
67          return STATE.CONTINUE;
68      }
69  
70      /**
71       * {@inheritDoc}
72       */
73      @Override
74      public AsyncHandler.STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception {
75          if (handshakeOccured.get()) {
76              // onOpen only called once
77              if (protocolEnabled && ++count == 1) {
78                  status = Socket.STATUS.INIT;
79              }
80              return super.onStatusReceived(responseStatus);
81          }
82          return STATE.CONTINUE;
83      }
84  
85      @Override
86      public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception {
87          handshakeOccured.set(true);
88          if (isBinary) {
89              byte[] payload = bodyPart.getBodyPartBytes();
90              if (protocolEnabled && !protocolReceived) {
91                  if (!Utils.whiteSpace(payload)) {
92                      TransportsUtil.invokeFunction(decoders, functions, payload.getClass(), payload, MESSAGE.name(), resolver);
93                      protocolReceived = true;
94                  }
95                  return AsyncHandler.STATE.CONTINUE;
96              } else if (!Utils.whiteSpace(payload)) {
97                  TransportsUtil.invokeFunction(decoders, functions, payload.getClass(), payload, MESSAGE.name(), resolver);
98              }
99              unlockFuture();
100         } else {
101             String m = new String(bodyPart.getBodyPartBytes(), charSet).trim();
102             if (protocolEnabled && !protocolReceived) {
103                 if (m.length() > 0) {
104                     TransportsUtil.invokeFunction(decoders, functions, m.getClass(), m, MESSAGE.name(), resolver);
105                     protocolReceived = true;
106                 }
107                 return AsyncHandler.STATE.CONTINUE;
108             } else if (m.length() > 0) {
109                 TransportsUtil.invokeFunction(decoders, functions, m.getClass(), m, MESSAGE.name(), resolver);
110             }
111             unlockFuture();
112         }
113         return AsyncHandler.STATE.CONTINUE;
114     }
115 
116     /**
117      * {@inheritDoc}
118      */
119     @Override
120     public Request.TRANSPORT name() {
121         return Request.TRANSPORT.LONG_POLLING;
122     }
123 
124 }
