16  package org.atmosphere.wasync.transport;
17  
18  
19  import com.ning.http.client.AsyncHttpClient;
20  import com.ning.http.client.HttpResponseStatus;
21  import com.ning.http.client.RequestBuilder;
22  import org.atmosphere.wasync.Decoder;
23  import org.atmosphere.wasync.FunctionResolver;
24  import org.atmosphere.wasync.FunctionWrapper;
25  import org.atmosphere.wasync.Request;
26  
27  import java.util.List;
28  import java.util.concurrent.atomic.AtomicBoolean;
29  
30  public class LongPollingTransport<T> extends StreamTransport {
31  
32      private final Request request;
33      private final AsyncHttpClient asyncHttpClient;
34      private final AtomicBoolean hasSucceedOnce = new AtomicBoolean(false);
35  
36      public LongPollingTransport(List<Decoder<?,?>>decoders, List<FunctionWrapper> functions, Request request, AsyncHttpClient asyncHttpClient, FunctionResolver resolver) {
37          super(decoders, functions, resolver);
38          this.request = request;
39          this.asyncHttpClient = asyncHttpClient;
40      }
41  
42      @Override
43      public Request.TRANSPORT name() {
44          return Request.TRANSPORT.LONG_POLLING;
45      }
46  
47      /**
48       * {@inheritDoc}
49       */
50      @Override
51      public STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception {
52          hasSucceedOnce.set(true);
53          return super.onStatusReceived(responseStatus);
54      }
55  
56      @Override
57      public String onCompleted() throws Exception {
58          if (hasSucceedOnce.get()) {
59              RequestBuilder r = new RequestBuilder();
60              r.setUrl(request.uri())
61                      .setMethod(request.method().name())
62                      .setHeaders(request.headers());
63  
64              java.util.concurrent.Future<String> s = asyncHttpClient.prepareRequest(r.build()).execute(this);
65          }
66  
67          return "";
68      }
69  }
