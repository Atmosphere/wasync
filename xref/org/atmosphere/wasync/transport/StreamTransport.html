37  public class StreamTransport<T> extends AbstractAsyncHandler<String> implements Transport {
38      private final static String DEFAULT_CHARSET = "UTF-8";
39      
40  
41      protected Future f;
42      protected final List<FunctionWrapper> functions;
43      private final List<Decoder<? extends Object, ?>> decoders;
44      //TODO fix me
45      protected String charSet = DEFAULT_CHARSET;
46      private final FunctionResolver resolver;
47      private final Options options;
48  
49      public StreamTransport(Options options, List<Decoder<? extends Object, ?>> decoders, List<FunctionWrapper> functions, FunctionResolver resolver) {
50          if (decoders.size() == 0) {
51              decoders.add(new Decoder<String, Object>() {
52                  @Override
53                  public Object decode(String s) {
54                      return s;
55                  }
56              });
57          }
58          this.decoders = decoders;
59          this.functions = functions;
60          this.resolver = resolver;
61          this.options = options;
62      }
63  
64      @Override
65      public Transport future(Future f) {
66          this.f = f;
67          return this;
68      }
69  
70      @Override
71      public Transport registerF(FunctionWrapper function) {
72          functions.add(function);
73          return this;
74      }
75  
76      /**
77       * {@inheritDoc}
78       */
79      @Override
80      public void onThrowable(Throwable t) {
81          TransportsUtil.invokeFunction(decoders, functions, t.getClass(), t, Function.MESSAGE.error.name(), resolver);
82      }
83  
84      /**
85       * {@inheritDoc}
86       */
87      @Override
88      public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception {
89          String m = new String(bodyPart.getBodyPartBytes(), charSet).trim();
90          if (!m.isEmpty()) {
91          	TransportsUtil.invokeFunction(decoders, functions, m.getClass(), m, Function.MESSAGE.message.name(), resolver);
92          }
93          return STATE.CONTINUE;
94      }
95  
96      /**
97       * {@inheritDoc}
98       */
99      @Override
100     public STATE onHeadersReceived(HttpResponseHeaders headers) throws Exception {
101         TransportsUtil.invokeFunction(decoders, functions, Map.class, headers.getHeaders(), Function.MESSAGE.headers.name(), resolver);
102 
103         // TODO: Parse charset
104         return STATE.CONTINUE;
105     }
106 
107     /**
108      * {@inheritDoc}
109      */
110     @Override
111     public STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception {
112         f.done();
113         TransportsUtil.invokeFunction(decoders, functions, String.class, "Open", Function.MESSAGE.open.name(), resolver);
114         TransportsUtil.invokeFunction(decoders, functions, Integer.class, new Integer(responseStatus.getStatusCode()), Function.MESSAGE.status.name(), resolver);
115 
116         return STATE.CONTINUE;
117     }
118 
119     @Override
120     public String onCompleted() throws Exception {
121         return "";
122     }
123 
124     @Override
125     public Request.TRANSPORT name() {
126         return Request.TRANSPORT.STREAMING;
127     }
128 
129     @Override
130     public void close() {
131         TransportsUtil.invokeFunction(decoders, functions, String.class, "Close", Function.MESSAGE.open.name(), resolver);
132     }

139 	@Override
140 	public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart, StringBuilder messagesStringBuilder, AtmosphereRequest atmosphereRequest) throws Exception {
141 		String m = new String(bodyPart.getBodyPartBytes(), charSet).trim();
142 		
143 		if(m.isEmpty()) {
144 			return STATE.CONTINUE;
145 		}
146 
147 		messagesStringBuilder.append(m);
148 		List<String> messages = atmosphereClientSideHandler(messagesStringBuilder, atmosphereRequest);
149 		if(messages==null) {
150 			return STATE.CONTINUE;
151 		}
152 		Iterator<String> it = messages.iterator();
153 	    while(it.hasNext()) {
154 	    	String message = it.next();
155 	        TransportsUtil.invokeFunction(decoders, functions, m.getClass(), message, Function.MESSAGE.message.name(), resolver);
156 	    }
157 	   return STATE.CONTINUE;
158 	}
