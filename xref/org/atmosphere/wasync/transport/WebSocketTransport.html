89      public WebSocketTransport(RequestBuilder requestBuilder, Options options, Request request, List<FunctionWrapper> functions) {
90          super();
91          this.decoders = request.decoders();
92  
93          if (decoders.size() == 0) {
94              decoders.add(new Decoder<String, Object>() {
95                  @Override
96                  public Object decode(Event e, String s) {
97                      return s;
98                  }
99              });
100         }
101         this.functions = functions;
102         this.resolver = request.functionResolver();
103         this.options = options;
104         this.requestBuilder = requestBuilder;
105         this.supportBinary = options.binary() ||
106                 // Backward compatibility.
107                 (request.headers().get("Content-Type") != null ?
108                         request.headers().get("Content-Type").contains("application/octet-stream") : false);
109 
110         protocolEnabled = request.queryString().get("X-atmo-protocol") != null;
111         timer = Executors.newSingleThreadScheduledExecutor();
112     }

127     @Override
128     public void close() {
129         status = Socket.STATUS.CLOSE;
130         if (closed.getAndSet(true)) return;
131 
132         if (options.reconnectInSeconds() <= 0 && !options.reconnect()) {
133             timer.shutdown();
134         }
135 
136         TransportsUtil.invokeFunction(CLOSE, decoders, functions, String.class, CLOSE.name(), CLOSE.name(), resolver);
137 
138         if (webSocket != null && webSocket.isOpen())
139             webSocket.close();
140 
141         futureDone();
142     }

199     @Override
200     public STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception {
201         logger.trace("Status received {}", responseStatus);
202         TransportsUtil.invokeFunction(STATUS, decoders, functions, Integer.class, new Integer(responseStatus.getStatusCode()), STATUS.name(), resolver);
203         if (responseStatus.getStatusCode() == 101) {
204             return STATE.UPGRADE;
205         } else {
206             logger.debug("Invalid status code {} for WebSocket Handshake", responseStatus.getStatusCode());
207             status = Socket.STATUS.ERROR;
208             throw new TransportNotSupported(responseStatus.getStatusCode(), responseStatus.getStatusText());
209         }
210     }

238     void unlockFuture() {
239         try {
240             connectOperationFuture.finishOrThrowException();
241         } catch (IOException e) {
242             logger.warn("", e);
243         }

246     /**
247      * {@inheritDoc}
248      */
249     @Override
250     public void onSuccess(WebSocket webSocket) {
251         logger.trace("onSuccess {}", webSocket);
252 
253         this.webSocket = webSocket;
254 
255         if (connectOperationFuture != null && !protocolEnabled) {
256             unlockFuture();
257         }
258 
259         ok.set(true);
260         WebSocketListener l = new TextListener();
261         if (supportBinary) {
262             l = new BinaryListener(l);
263         }
264         webSocket.addWebSocketListener(l);
265         l.onOpen(webSocket);
266     }

268     void connectFutureException(Throwable t) {
269         IOException e = IOException.class.isAssignableFrom(t.getClass()) ? IOException.class.cast(t) : new IOException(t);
270         connectOperationFuture.ioException(e).done();
271     }

274     void tryReconnect() {
275 
276         reconnectAttempt.incrementAndGet();
277 
278         if (options.reconnectInSeconds() > 0) {
279             timer.schedule(new Runnable() {
280                 public void run() {
281                     reconnect();
282                 }
283             }, options.reconnectInSeconds(), TimeUnit.SECONDS);
284         } else {
285             reconnect();
286         }
287 
288     }
289 
290     void reconnect() {
291         try {
292             reconnecting.set(true);
293             status = Socket.STATUS.REOPENED;
294 
295             ListenableFuture<WebSocket> webSocketListenableFuture = options.runtime().executeRequest(requestBuilder.build(), WebSocketTransport.this);
296 
297             logger.info("try reconnect : attempt [{}/{}]", reconnectAttempt.get(), options.reconnectAttempts());
298 
299             webSocketListenableFuture.get();
300 
301             logger.info("reconnect successful ! in attempt [{}/{}]", reconnectAttempt.get(), options.reconnectAttempts());
302 
303             reconnectAttempt.set(0);
304             reconnecting.set(false);
305         } catch (InterruptedException e) {
306             reconnecting.set(false);
307             logger.error("", e);
308         } catch (ExecutionException e) {
309 
310             if (reconnectAttempt.get() < options.reconnectAttempts()) {
311                 tryReconnect();
312             } else {
313                 reconnecting.set(false);
314                 reconnectAttempt.set(0);
315                 onFailure(e.getCause() != null ? e.getCause() : e);
316             }
317         }
318     }

368     private final class TextListener implements WebSocketTextListener {
369         @Override
370         public void onMessage(String message) {
371             logger.trace("onMessage {} for {}", message, webSocket);
372             message = message.trim();
373             logger.trace("{} received {}", name(), message);
374             if (message.length() > 0) {
375                 TransportsUtil.invokeFunction(MESSAGE,
376                         decoders,
377                         functions,
378                         message.getClass(),
379                         message,
380                         MESSAGE.name(),
381                         resolver);
382 
383                 // Since the protocol is enabled, handshake occurred, now ready so go asynchronous
384                 if (connectOperationFuture != null && protocolEnabled) {
385                     unlockFuture();
386                 }
387             }
388         }
389 
390         @Override
391         public void onOpen(WebSocket websocket) {
392             logger.trace("onOpen for {}", webSocket);
393 
394             // Could have been closed during the handshake.
395             if (status.equals(Socket.STATUS.CLOSE) || status.equals(Socket.STATUS.ERROR)) return;
396 
397             closed.set(false);
398             Event newStatus = status.equals(Socket.STATUS.INIT) ? OPEN : REOPENED;
399             status = Socket.STATUS.OPEN;
400             TransportsUtil.invokeFunction(newStatus,
401                     decoders, functions, String.class, newStatus.name(), newStatus.name(), resolver);
402         }
403 
404         @Override
405         public void onClose(WebSocket websocket) {
406             logger.trace("onClose for {}", webSocket);
407             if (closed.get()) return;
408 
409             close();
410             if (options.reconnect()) {
411                 tryReconnect();
412             }
413         }
414 
415         @Override
416         public void onError(Throwable t) {
417             logger.trace("onError for {}", t);
418             status = Socket.STATUS.ERROR;
419             logger.debug("", t);
420             onFailure(t);
421         }
422     }
