53  public class WebSocketTransport extends WebSocketUpgradeHandler implements Transport {
54  
55      private final Logger logger = LoggerFactory.getLogger(WebSocketTransport.class);
56      private WebSocket webSocket;
57      private final AtomicBoolean ok = new AtomicBoolean(false);
58      private final List<FunctionWrapper> functions;
59      private final List<Decoder<?, ?>> decoders;
60      private final FunctionResolver resolver;
61      private final Options options;
62      private final RequestBuilder requestBuilder;
63      private final AtomicBoolean closed = new AtomicBoolean(false);
64      private STATUS status = Socket.STATUS.INIT;
65      private final AtomicBoolean errorHandled = new AtomicBoolean();
66  
67      public WebSocketTransport(RequestBuilder requestBuilder, Options options, Request request, List<FunctionWrapper> functions) {
68          super(new Builder());
69          this.decoders = request.decoders();
70  
71          if (decoders.size() == 0) {
72              decoders.add(new Decoder<String, Object>() {
73                  @Override
74                  public Object decode(Event e, String s) {
75                      return s;
76                  }
77              });
78          }
79          this.functions = functions;
80          this.resolver = request.functionResolver();
81          this.options = options;
82          this.requestBuilder = requestBuilder;
83      }
84  
85      /**
86       * {@inheritDoc}
87       */
88      @Override
89      public void onThrowable(Throwable t) {
90          status = Socket.STATUS.ERROR;
91          errorHandled.set(TransportsUtil.invokeFunction(Event.ERROR, decoders, functions, t.getClass(), t, ERROR.name(), resolver));
92      }
93  
94      /**
95       * {@inheritDoc}
96       */
97      @Override
98      public void close() {
99          if (closed.getAndSet(true)) return;
100 
101         status = Socket.STATUS.CLOSE;
102 
103         TransportsUtil.invokeFunction(CLOSE, decoders, functions, String.class, CLOSE.name(), CLOSE.name(), resolver);
104 
105         if (webSocket != null)
106             webSocket.close();
107     }

125     @Override
126     public void error(Throwable t) {
127         logger.warn("", t);

129     }

131     /**
132      * {@inheritDoc}
133      */
134     @Override
135     public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception {
136         return STATE.CONTINUE;
137     }

139     /**
140      * {@inheritDoc}
141      */
142     @Override
143     public STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception {
144         TransportsUtil.invokeFunction(STATUS, decoders, functions, Integer.class, new Integer(responseStatus.getStatusCode()), STATUS.name(), resolver);
145         if (responseStatus.getStatusCode() == 101) {
146             return STATE.UPGRADE;
147         } else {
148             status = Socket.STATUS.ERROR;
149             throw new TransportNotSupported(responseStatus.getStatusCode(), responseStatus.getStatusText());
150         }
151     }
152 
153     /**
154      * {@inheritDoc}
155      */
156     @Override
157     public STATE onHeadersReceived(HttpResponseHeaders headers) throws Exception {
158         TransportsUtil.invokeFunction(HEADERS, decoders, functions, Map.class, headers.getHeaders(), HEADERS.name(), resolver);
159 
160         return STATE.CONTINUE;
161     }

163     /**
164      * {@inheritDoc}
165      */
166     @Override
167     public WebSocket onCompleted() throws Exception {
168         if (webSocket == null) {
169             status = Socket.STATUS.ERROR;
170             return null;
171         }
172         TransportsUtil.invokeFunction(TRANSPORT, decoders, functions, Request.TRANSPORT.class, name(), TRANSPORT.name(), resolver);
173         return webSocket;
174     }
175 
176     /**
177      * {@inheritDoc}
178      */
179     @Override
180     public void onSuccess(WebSocket webSocket) {
181         this.webSocket = webSocket;
182 
183         ok.set(true);
184         WebSocketTextListener l = new WebSocketTextListener() {
185             @Override
186             public void onMessage(String message) {
187                 message = message.trim();
188                 logger.debug("{} received {}", name(), message );
189                 if (message.length() > 0) {
190                     TransportsUtil.invokeFunction(MESSAGE,
191                             decoders,
192                             functions,
193                             message.getClass(),
194                             message,
195                             MESSAGE.name(),
196                             resolver);
197                 }
198             }
199 
200             @Override
201             public void onFragment(String fragment, boolean last) {
202             }
203 
204             @Override
205             public void onOpen(WebSocket websocket) {
206                 // Could have been closed during the handshake.
207                 if (status.equals(Socket.STATUS.CLOSE)) return;
208 
209                 boolean reconnect = false;
210                 if (!status.equals(Socket.STATUS.INIT)) {
211                     reconnect = true;
212                 }
213 
214                 status = Socket.STATUS.OPEN;
215 
216                 TransportsUtil.invokeFunction(reconnect ? RECONNECT : OPEN,
217                         decoders, functions, String.class, OPEN.name(), OPEN.name(), resolver);
218             }
219 
220             @Override
221             public void onClose(WebSocket websocket) {
222                 if (closed.get()) return;
223 
224                 status = Socket.STATUS.INIT;
225 
226                 TransportsUtil.invokeFunction(CLOSE, decoders, functions, String.class, CLOSE.name(), CLOSE.name(), resolver);
227                 if (options.reconnect()) {
228                     if (options.reconnectInSeconds() > 0) {
229                         ScheduledExecutorService e = options.runtime().getConfig().reaper();
230                         e.schedule(new Runnable() {
231                             public void run() {
232                                 reconnect();
233                             }
234                         }, options.reconnectInSeconds(), TimeUnit.SECONDS);
235                     } else {
236                         reconnect();
237                     }
238                 } else {
239                     status = Socket.STATUS.CLOSE;
240                 }
241             }
242 
243             @Override
244             public void onError(Throwable t) {
245                 status = Socket.STATUS.ERROR;
246 
247                 errorHandled.set(TransportsUtil.invokeFunction(ERROR, decoders, functions, t.getClass(), t, ERROR.name(), resolver));
248             }
249         };
250         webSocket.addWebSocketListener(l);
251         l.onOpen(webSocket);
252     }
253 
254     void reconnect() {
255         try {
256             options.runtime().executeRequest(requestBuilder.build(), WebSocketTransport.this);
257         } catch (IOException e) {
258             logger.error("", e);
259         }
260     }

285     }
