16  package org.atmosphere.wasync;
17  
18  import com.ning.http.client.AsyncHttpClient;
19  
20  /**
21   * Base class for building {@link Options}
22   *
23   * @author Jeanfrancois Arcand
24   * @param <T>
25   */
26  public abstract class OptionsBuilder<T extends OptionsBuilder<T>> {
27  
28      private Transport transport;
29      private boolean reconnect = true;
30      private int reconnectInSecond = 0;
31      private long waitBeforeUnlocking = 2000;
32      private AsyncHttpClient client;
33      private boolean runtimeShared = false;
34      private int requestTimeout = -1;
35      protected final Class<T> derived;
36  
37      protected OptionsBuilder(Class<T> derived) {
38          this.derived = derived;
39      }
40  
41      /**
42       * The time, in seconds, the connection will stay open when waiting for new messages. This can be seen as the idle time.
43       *
44       * @return this
45       */
46      public T requestTimeoutInSeconds(int requestTimeout) {
47          this.requestTimeout = requestTimeout;
48          return derived.cast(this);
49      }
50  
51      /**
52       * Register a new {@link Transport} implementation. Register a transport only if you are planning to use
53       * a different transport than the supported one.
54       *
55       * @param transport {@link Transport}
56       * @return this
57       */
58      public T registerTransport(Transport transport) {
59          this.transport = transport;
60          return derived.cast(this);
61      }
62  
63      /**
64       * Automatically reconnect in case the connection get closed. Default is <tt>true</tt>
65       *
66       * @param reconnect reconnect in case the connection get closed. Default is <tt>true</tt>
67       * @return this
68       */
69      public T reconnect(boolean reconnect) {
70          this.reconnect = reconnect;
71          return derived.cast(this);
72      }
73  
74      /**
75       * The time in second, before the reconnection occurs. Default is 1 second
76       *
77       * @param reconnectInSecond time in second, before the reconnection occurs. Default is 1 second
78       * @return this
79       */
80      public T pauseBeforeReconnectInSeconds(int reconnectInSecond) {
81          this.reconnectInSecond = reconnectInSecond;
82          return derived.cast(this);
83      }
84  
85      /**
86       * For streaming and long-polling, the server may not send the headers so the client never knows
87       * if the connection succeeded or not. By default the library will wait for 2500 milliseconds before
88       * considering the connection established.
89       *
90       * @param waitBeforeUnlocking the time in millisecond
91       * @return this
92       */
93      public T waitBeforeUnlocking(long waitBeforeUnlocking) {
94          this.waitBeforeUnlocking = waitBeforeUnlocking;
95          return derived.cast(this);
96      }
97  
98      /**
99       * Set to true if your AsyncHttpClient is shared between clients.
100      * by the library.
101      *
102      * @param runtimeShared true if your AsyncHttpClient is shared between clients.
103      * @return this;
104      */
105     public T runtimeShared(boolean runtimeShared) {
106         this.runtimeShared = runtimeShared;
107         return derived.cast(this);
108     }
109 
110     /**
111      * Allow an application that want to share {@link AsyncHttpClient} or configure it before it gets used
112      * by the library.
113      *
114      * @param client
115      * @return this;
116      */
117     public T runtime(AsyncHttpClient client) {
118         return runtime(client, false);
119     }
120 
121     /**
122      * Allow an application that want to share {@link AsyncHttpClient} or configure it before it gets used
123      * by the library.
124      *
125      * @param client
126      * @param runtimeShared to true if the runtime is shared between clients. If shared, the asyncHttpClient.close()
127      *                      must be invoked by the application itself.
128      * @return this;
129      */
130     public T runtime(AsyncHttpClient client, boolean runtimeShared) {
131         this.client = client;
132         this.runtimeShared = runtimeShared;
133         return derived.cast(this);
134     }
135 
136     /**
137      * Build an {@link Options}
138      *
139      * @return {@link Options}
140      */
141     public abstract Options build();
142 
143     /**
144      * The used {@link Transport}
145      * @return {@link Transport}
146      */
147     public Transport transport() {
148         return transport;
149     }
150     /**
151      * Reconnect after a network failure or when the server close the connection.
152      * @return reconnect
153      */
154     public boolean reconnect(){
155         return reconnect;
156     }
157     /**
158      * The delay, in second, before reconnecting.
159      * @return The delay, in second, before reconnecting.
160      */
161     public int reconnectInSeconds(){
162         return reconnectInSecond;
163     }
164     /**
165      * When using long-polling and the {@link Request}, the delay before considering the long-polling connection has been fully processed by the server. If you use
166      * the {@link org.atmosphere.wasync.impl.AtmosphereClient}, the server will send some handshake so this value is not needed.
167      * @return the delay before considering the long-polling connection has been fully processed
168      */
169     public long waitBeforeUnlocking() {
170         return waitBeforeUnlocking;
171     }
172     /**
173      * The {@link AsyncHttpClient} used to communicate with server.
174      * @return {@link AsyncHttpClient} used to communicate with server.
175      */
176     public AsyncHttpClient runtime(){
177         return client;
178     }
179     /**
180      * Return true is the {@link AsyncHttpClient} is shared between {@link Socket}. Default is false. You need to invoke {@link #runtime(com.ning.http.client.AsyncHttpClient)} to make
181      * it shared.
182      * @return true is the {@link AsyncHttpClient}
183      */
184     public boolean runtimeShared(){
185         return runtimeShared;
186     }
187     /**
188      * The time, in seconds to wait before closing the connection.
189      * @return The time, in seconds to wait before closing the connection.
190      */
191     public int requestTimeoutInSeconds() {
192         return requestTimeout;
193     }
194 
195 }
