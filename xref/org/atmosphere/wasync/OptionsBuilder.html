16  package org.atmosphere.wasync;

18  import com.ning.http.client.AsyncHttpClient;
19  
20  /**
21   * Base class for building {@link Options}
22   *
23   * @author Jeanfrancois Arcand
24   * @param <T>
25   */
26  public abstract class OptionsBuilder<U extends Options, T extends OptionsBuilder<U,T>> {
27  
28      private Transport transport;
29      private boolean reconnect = true;
30      private int reconnectInSecond = 0;
31      private int reconnectAttempts = 0;
32      private long waitBeforeUnlocking = 2000;
33      private AsyncHttpClient client;
34      private boolean runtimeShared = false;
35      private int requestTimeout = -1;
36      protected final Class<T> derived;
37      private boolean binary;
38  
39      protected OptionsBuilder(Class<T> derived) {
40          this.derived = derived;
41      }
42  
43      /**
44       * The time, in seconds, the connection will stay open when waiting for new messages. This can be seen as the idle time.
45       *
46       * @return this
47       */
48      public T requestTimeoutInSeconds(int requestTimeout) {
49          this.requestTimeout = requestTimeout;
50          return derived.cast(this);
51      }
52  
53      /**
54       * Register a new {@link Transport} implementation. Register a transport only if you are planning to use
55       * a different transport than the supported one.
56       *
57       * @param transport {@link Transport}
58       * @return this
59       */
60      public T registerTransport(Transport transport) {
61          this.transport = transport;
62          return derived.cast(this);
63      }
64  
65      /**
66       * Automatically reconnect in case the connection get closed. Default is <tt>true</tt>
67       *
68       * @param reconnect reconnect in case the connection get closed. Default is <tt>true</tt>
69       * @return this
70       */
71      public T reconnect(boolean reconnect) {
72          this.reconnect = reconnect;
73          return derived.cast(this);
74      }
75  
76      /**
77       * The time in second, before the reconnection occurs. Default is 1 second
78       *
79       * @param reconnectInSecond time in second, before the reconnection occurs. Default is 1 second
80       * @return this
81       */
82      public T pauseBeforeReconnectInSeconds(int reconnectInSecond) {
83          this.reconnectInSecond = reconnectInSecond;
84          return derived.cast(this);
85      }
86  
87  
88      /**
89       * Maximum reconnection attempts that will be executed if the connection is lost. Must be used in conjunction with {@link #reconnectInSeconds}
90       */
91      public T reconnectAttempts(int reconnectAttempts) {
92          this.reconnectAttempts = reconnectAttempts;
93          return derived.cast(this);
94      }
95  
96      /**
97       * For streaming and long-polling, the server may not send the headers so the client never knows
98       * if the connection succeeded or not. By default the library will wait for 2500 milliseconds before
99       * considering the connection established.
100      *
101      * @param waitBeforeUnlocking the time in millisecond
102      * @return this
103      */
104     public T waitBeforeUnlocking(long waitBeforeUnlocking) {
105         this.waitBeforeUnlocking = waitBeforeUnlocking;
106         return derived.cast(this);
107     }
108 
109     /**
110      * Set to true if your AsyncHttpClient is shared between clients.
111      * by the library.
112      *
113      * @param runtimeShared true if your AsyncHttpClient is shared between clients.
114      * @return this;
115      */
116     public T runtimeShared(boolean runtimeShared) {
117         this.runtimeShared = runtimeShared;
118         return derived.cast(this);
119     }
120 
121     /**
122      * Allow an application that want to share {@link AsyncHttpClient} or configure it before it gets used
123      * by the library. Application must call {@link com.ning.http.client.AsyncHttpClient#close()} at the end of their
124      * execution as the library won't close it since shared.
125      *
126      * @param client
127      * @return this;
128      */
129     public T runtime(AsyncHttpClient client) {
130         return runtime(client, true);
131     }
132 
133     /**
134      * Allow an application that want to share {@link AsyncHttpClient} or configure it before it gets used
135      * by the library.
136      *
137      * @param client
138      * @param runtimeShared to true if the runtime is shared between clients. If shared, the asyncHttpClient.close()
139      *                      must be invoked by the application itself.
140      * @return this;
141      */
142     public T runtime(AsyncHttpClient client, boolean runtimeShared) {
143         this.client = client;
144         this.runtimeShared = runtimeShared;
145         return derived.cast(this);
146     }
147 
148     /**
149      * Build an {@link Options}
150      *
151      * @return {@link Options}
152      */
153     public abstract U build();
154 
155     /**
156      * The used {@link Transport}
157      * @return {@link Transport}
158      */
159     public Transport transport() {
160         return transport;
161     }
162     /**
163      * Reconnect after a network failure or when the server close the connection.
164      * @return reconnect
165      */
166     public boolean reconnect(){
167         return reconnect;
168     }
169     /**
170      * The delay, in second, before reconnecting.
171      * @return The delay, in second, before reconnecting.
172      */
173     public int reconnectInSeconds(){
174         return reconnectInSecond;
175     }
176 
177     /**
178      * Maximum reconnection attempts that will be executed if the connection is lost. Must be used in conjunction with {@link #reconnectInSeconds}
179      *
180      * @return the number of maximum reconnection attempts
181      */
182     public int reconnectAttempts(){
183         return reconnectAttempts;
184     }
185     /**
186      * The delay before considering the http connection has been fully processed by the server. By default, the library will wait 2 seconds before allowing the {@link Socket#fire(Object)}
187      * to send message. Server side framework that aren't sending any data when suspending a connection may not be ready to fullfil request, hence some data may be lost.
188      *
189      * Framework like Atmosphere, Socket.io, Cometd do send some "handshake data" allow the fire operation to be available fast without loosing ant data.
190      */
191     public long waitBeforeUnlocking() {
192         return waitBeforeUnlocking;
193     }
194     /**
195      * The {@link AsyncHttpClient} used to communicate with server.
196      * @return {@link AsyncHttpClient} used to communicate with server.
197      */
198     public AsyncHttpClient runtime(){
199         return client;
200     }
201     /**
202      * Return true is the {@link AsyncHttpClient} is shared between {@link Socket}. Default is false. You need to invoke {@link #runtime(com.ning.http.client.AsyncHttpClient)} to make
203      * it shared.
204      * @return true is the {@link AsyncHttpClient}
205      */
206     public boolean runtimeShared(){
207         return runtimeShared;
208     }
209     /**
210      * The time, in seconds to wait before closing the connection.
211      * @return The time, in seconds to wait before closing the connection.
212      */
213     public int requestTimeoutInSeconds() {
214         return requestTimeout;
215     }
216 
217     /**
218      * Set to true in order to received byte data from the server. By default, binary aren't supported and response's server are considered as String.
219      * @param binary true to enabled
220      * @return this;
221      */
222     public T binary(boolean binary) {
223         this.binary = binary;
224         return derived.cast(this);
225     }
226 
227     /**
228      * Return true is server's response binary is supported. Default is false
229      * @return true is server's response binary is supported. Default is false
230      */
231     public boolean binary() {
232         return binary;
233     }
