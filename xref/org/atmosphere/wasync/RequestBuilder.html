16  package org.atmosphere.wasync;
17  
18  import java.util.ArrayList;
19  import java.util.Collection;
20  import java.util.HashMap;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.concurrent.CopyOnWriteArrayList;
24  
25  /**
26   * Base class for building {@link Request}
27   *
28   * @author jeanfrancois Arcand
29   */
30  public abstract class RequestBuilder<T extends RequestBuilder<T>> {
31  
32      protected final List<Request.TRANSPORT> transports = new ArrayList<Request.TRANSPORT>();
33      protected Request.METHOD method = Request.METHOD.GET;
34      protected String uri = "http://localhost:8080";
35      protected final List<Encoder<?, ?>> encoders = new CopyOnWriteArrayList<Encoder<?, ?>>();
36      protected final List<Decoder<?, ?>> decoders = new CopyOnWriteArrayList<Decoder<?, ?>>();
37      protected final Map<String, Collection<String>> headers = new HashMap<String, Collection<String>>();
38      protected final Map<String, List<String>> queryString = new HashMap<String, List<String>>();
39      protected FunctionResolver resolver = FunctionResolver.DEFAULT;
40      protected final Class<T> derived;
41  
42      protected RequestBuilder(Class<T> derived) {
43          this.derived = derived;
44      }
45  
46      /**
47       * The {@link Request.TRANSPORT} to use. This method can be invoked several time and the library will loop over the list
48       * until one {@link Request.TRANSPORT} succeed. The first added is always the first used.
49       * @param t
50       * @return this
51       */
52      public T transport(Request.TRANSPORT t) {
53          transports.add(t);
54          return derived.cast(this);
55      }
56  
57      /**
58       * The method to use for connecting tho the remote server. It is recommended to always use {@link Request.METHOD#GET}
59       * @param method
60       * @return this
61       */
62      public T method(Request.METHOD method) {
63          this.method = method;
64          return derived.cast(this);
65      }
66  
67      /**
68       * The URI to connect to.
69       * @param uri  a uri to connect to
70       * @return this
71       */
72      public T uri(String uri) {
73          this.uri = uri;
74          return derived.cast(this);
75      }
76  
77      /**
78       * Add an {@link Encoder}. Several Encoder can be added and will be invoked the order they were added.
79       * @param e an {@link Encoder}
80       * @return this
81       */
82      public T encoder(Encoder e) {
83          encoders.add(e);
84          return derived.cast(this);
85      }
86  
87      /**
88       * Add a {@link Decoder}. Several Decoder can be added and will be invoked the order they were added.
89       * @param d a {@link Decoder}
90       * @return this
91       */
92      public T decoder(Decoder d) {
93          decoders.add(d);
94          return derived.cast(this);
95      }
96  
97      /**
98       * Add a header.
99       * @param name header name
100      * @param value header value
101      * @return this
102      */
103     public T header(String name, String value) {
104         Collection<String> l = headers.get(name);
105         if (l == null) {
106             l = new ArrayList<String>();
107         }
108         l.add(value);
109         headers.put(name, l);
110         return derived.cast(this);
111     }
112 
113     /**
114      * Add a query param.
115      * @param name header name
116      * @param value header value
117      * @return this
118      */
119     public T queryString(String name, String value) {
120         List<String> l = queryString.get(name);
121         if (l == null) {
122             l = new ArrayList<String>();
123         }
124         l.add(value);
125         queryString.put(name, l);
126         return derived.cast(this);
127     }
128 
129     /**
130      * Add a {@link FunctionResolver}
131      * @param resolver  a {@link FunctionResolver}
132      * @return this
133      */
134     public T resolver(FunctionResolver resolver) {
135         this.resolver = resolver;
136         return derived.cast(this);
137     }
138 
139     public abstract Request build();
140 
141     /**
142      * Return the current list of {@link Request.TRANSPORT}
143      * @return the current list of {@link Request.TRANSPORT}
144      */
145     public List<Request.TRANSPORT> transports() {
146         return transports;
147     }
148 
149     /**
150      * Return the HTTP method
151      * @return  the HTTP method
152      */
153     public Request.METHOD method() {
154         return method;
155     }
156 
157     /**
158      * Return the current tMap of headers
159      * @return the current tMap of headers
160      */
161     public Map<String, Collection<String>> headers() {
162         return headers;
163     }
164 
165     /**
166      * Return the current query string/form param
167      * @return  the current query string/form param
168      */
169     public Map<String, List<String>> queryString() {
170         return queryString;
171     }
172 
173     /**
174      * Return the current list of {@link Encoder}
175      * @return  the current list of {@link Encoder}
176      */
177     public List<Encoder<?, ?>> encoders() {
178         return encoders;
179     }
180 
181     /**
182      * Return the current list of {@link Decoder}
183      * @return  the current list of {@link Decoder}
184      */
185     public List<Decoder<?, ?>> decoders() {
186         return decoders;
187     }
188 
189     /**
190      * Return the uri
191      * @return the uri
192      */
193     public String uri() {
194         return uri;
195     }
196 
197     /**
198      * Return the current {@link FunctionResolver}
199      * @return the current {@link FunctionResolver}
200      */
201     public FunctionResolver resolver() {
202         return resolver;
203     }
204 }
