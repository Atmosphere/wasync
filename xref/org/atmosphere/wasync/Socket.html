16  package org.atmosphere.wasync;
17  
18  import java.io.IOException;
19  
20  /**
21   * A Socket represent a connection to a remote server. A Socket abstract the transport used, the client will negotiate
22   * the best {@link org.atmosphere.wasync.Request#transport()} to communicate with the remote Server. As simple as
23   * <blockquote><pre>
24       Client client = AtmosphereClientFactory.getDefault().newclient();
25  
26       RequestBuilder request = client.newRequestBuilder()
27               .method(Request.METHOD.GET)
28               .uri(targetUrl + "/suspend")
29               .encoder(new Encoder&lt;String, Reader&gt;() {        // Stream the request body
30                   &#64;Override
31                   public Reader encode(String s) {
32                       return new StringReader(s);
33                   }
34               })
35               .decoder(new Decoder&lt;String, Reader&gt;() {
36                    &#64;Override
37                   public Reader decode(String s) {
38                       return new StringReader(s);
39                   }
40               })
41               .transport(WEBSOCKET)                        // Try WebSocket
42               .transport(LONG_POLLING);                    // Fallback to Long-Polling
43  
44       Socket socket = client.create();
45       socket.on("message", new Function&lt;String&gt;() {
46           &#64;Override
47           public void on(Reader r) {
48               // Read the response
49           }
50       }).on(new Function&lt;IOException&gt;() {
51  
52           &#64;Override
53           public void on(Throwable t) {
54               // Some IOException occurred
55           }
56  
57       }).open(request.build()).fire("echo");
58   * </pre></blockquote>
59   *
60   * @author Jeanfrancois Arcand
61   */
62  public interface Socket {
63  
64      enum EVENT {CLOSE, ERROR, HEADER, STATUS, MESSAGE}
65  
66      /**
67       * Send data to the remote Server.
68       * @param data
69       * @return a {@link Future}
70       * @throws IOException
71       */
72      Future fire(Object data) throws IOException;
73  
74      /**
75       * Associate a {@link Function} with the Socket. When a response is received, the library will try to associated
76       * the decoded message (decoded by {@link Decoder}) to the defined type of the {@link Function}
77       * @param function a {@link Function}
78       * @return this
79       */
80      Socket on(Function<? extends Object> function);
81  
82      /**
83       * Associate a {@link Function} with the Socket. When a response is received, the library will try to associated
84       * the decoded message (decoded by {@link Decoder}) to the defined type of the {@link Function}. The default messages
85       * are defined by {@link Function.MESSAGE} but handling of custom message can be done using a {@link FunctionResolver}
86       * @param function a {@link Function}
87       * @return this
88       */
89      Socket on(String functionMessage, Function<? extends Object> function);
90  
91      /**
92       * Connect to the remote Server using the {@link Request}'s information.
93       * @param request a {@link Request}
94       * @return this
95       * @throws IOException
96       */
97      Socket open(Request request) throws IOException;
98  
99      /**
100      * Close this Socket
101      */
102     void close();
103 
104 }
